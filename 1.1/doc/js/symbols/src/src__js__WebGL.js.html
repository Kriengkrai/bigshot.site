 17 /**
 18  * Creates a new WebGL wrapper instance.
 19  *
 20  * @class WebGL wrapper for common {@link bigshot.VRPanorama} uses.
 21  * @param {HTMLCanvasElement} canvas_ the canvas
 22  * @see #onresize()
 23  */
 24 bigshot.WebGL = function (canvas_) {
 25     
 26     this.canvas = canvas_;
 27     
 28     this.gl = bigshot.webglutil.createContext (this.canvas); 
 29     
 30     /**
 31      * Must be called when the canvas element is resized.
 32      *
 33      * @public
 34      */
 35     this.onresize = function () {
 36         this.gl.viewportWidth = this.canvas.width;
 37         this.gl.viewportHeight = this.canvas.height;
 38     }
 39     
 40     this.onresize ();
 41     
 42     /**
 43      * Fragment shader. Taken from the "Learning WebGL" lessons:
 44      *     http://learningwebgl.com/blog/?p=571
 45      */
 46     this.fragmentShader = 
 47         "#ifdef GL_ES\n" + 
 48         "    precision highp float;\n" + 
 49         "#endif\n" + 
 50         "\n" + 
 51         "varying vec2 vTextureCoord;\n" + 
 52         "\n" + 
 53         "uniform sampler2D uSampler;\n" + 
 54         "\n" + 
 55         "void main(void) {\n" + 
 56         "    gl_FragColor = texture2D(uSampler, vec2(vTextureCoord.s, vTextureCoord.t));\n" + 
 57         "}\n";
 58     
 59     /**
 60      * Vertex shader. Taken from the "Learning WebGL" lessons:
 61      *     http://learningwebgl.com/blog/?p=571
 62      */
 63     this.vertexShader = 
 64         "attribute vec3 aVertexPosition;\n" +
 65         "attribute vec2 aTextureCoord;\n" +
 66         "\n" +
 67         "uniform mat4 uMVMatrix;\n" +
 68         "uniform mat4 uPMatrix;\n" +
 69         "\n" +
 70         "varying vec2 vTextureCoord;\n" +
 71         "\n" +
 72         "void main(void) {\n" +
 73         "    gl_Position = uPMatrix * uMVMatrix * vec4(aVertexPosition, 1.0);\n" +
 74         "    vTextureCoord = aTextureCoord;\n" +
 75         "}";
 76     
 77     /**
 78      * Creates a new shader.
 79      *
 80      * @type WebGLShader
 81      * @param {String} source the source code
 82      * @param {int} type the shader type, one of WebGLRenderingContext.FRAGMENT_SHADER or 
 83      * WebGLRenderingContext.VERTEX_SHADER
 84      */
 85     this.createShader = function (source, type) {
 86         var shader = this.gl.createShader (type);
 87         this.gl.shaderSource (shader, source);
 88         this.gl.compileShader (shader);
 89         
 90         if (!this.gl.getShaderParameter (shader, this.gl.COMPILE_STATUS)) {
 91             alert (this.gl.getShaderInfoLog (shader));
 92             return null;
 93         }
 94         
 95         return shader;
 96     };
 97     
 98     /**
 99      * Creates a new fragment shader.
100      *
101      * @type WebGLShader
102      * @param {String} source the source code
103      */
104     this.createFragmentShader = function (source) {
105         return this.createShader (source, this.gl.FRAGMENT_SHADER);
106     };
107     
108     /**
109      * Creates a new vertex shader.
110      *
111      * @type WebGLShader
112      * @param {String} source the source code
113      */
114     this.createVertexShader = function (source) {
115         return this.createShader (source, this.gl.VERTEX_SHADER);
116     };
117     
118     /**
119      * The current shader program.
120      */
121     this.shaderProgram = null;
122     
123     /**
124      * Initializes the shaders.
125      */
126     this.initShaders = function () {
127         this.shaderProgram = this.gl.createProgram ();
128         this.gl.attachShader (this.shaderProgram, this.createVertexShader (this.vertexShader));
129         this.gl.attachShader (this.shaderProgram, this.createFragmentShader (this.fragmentShader));
130         this.gl.linkProgram (this.shaderProgram);
131         
132         if (!this.gl.getProgramParameter (this.shaderProgram, this.gl.LINK_STATUS)) {
133             throw new Error ("Could not initialise shaders");
134             return;
135         }
136         
137         this.gl.useProgram (this.shaderProgram);
138         
139         this.shaderProgram.vertexPositionAttribute = this.gl.getAttribLocation (this.shaderProgram, "aVertexPosition");
140         this.gl.enableVertexAttribArray (this.shaderProgram.vertexPositionAttribute);
141         
142         this.shaderProgram.textureCoordAttribute = this.gl.getAttribLocation (this.shaderProgram, "aTextureCoord");
143         this.gl.enableVertexAttribArray (this.shaderProgram.textureCoordAttribute);
144         
145         this.shaderProgram.pMatrixUniform = this.gl.getUniformLocation(this.shaderProgram, "uPMatrix");
146         this.shaderProgram.mvMatrixUniform = this.gl.getUniformLocation(this.shaderProgram, "uMVMatrix");
147         this.shaderProgram.samplerUniform = this.gl.getUniformLocation(this.shaderProgram, "uSampler");
148     };
149     
150     /**
151      * The current object-to-world transform matrix.
152      *
153      * @type bigshot.TransformStack
154      */
155     this.mvMatrix = new bigshot.TransformStack ();
156     
157     this.pMatrix = new bigshot.TransformStack ();
158     
159     /**
160      * Sets the matrix parameters ("uniforms", since the variables are declared as uniform) in the shaders.
161      */
162     this.setMatrixUniforms = function () {
163         this.gl.uniformMatrix4fv(this.shaderProgram.pMatrixUniform, false, new Float32Array(this.pMatrix.matrix().flatten()));
164         this.gl.uniformMatrix4fv(this.shaderProgram.mvMatrixUniform, false, new Float32Array(this.mvMatrix.matrix().flatten()));
165     }
166     
167     /**
168      * Creates a texture from an image.
169      *
170      * @param {HTMLImageElement or HTMLCanvasElement} image the image
171      * @type WebGLTexture
172      * @return An initialized texture
173      */
174     this.createImageTextureFromImage = function (image, minFilter, magFilter) {
175         var texture = this.gl.createTexture();
176         this.handleImageTextureLoaded (this, texture, image, minFilter, magFilter);
177         return texture;
178     }
179     
180     /**
181      * Creates a texture from a source url.
182      *
183      * @param {String} source the URL of the image
184      * @return WebGLTexture
185      */
186     this.createImageTextureFromSource = function (source, minFilter, magFilter) {
187         var image = new Image();
188         var texture = this.gl.createTexture();
189         
190         var that = this;
191         image.onload = function () {
192             that.handleImageTextureLoaded (that, texture, image, minFilter, magFilter);
193         }
194         
195         image.src = source;
196         
197         return texture;
198     }
199     
200     /**
201      * Uploads the image data to the texture memory. Called when the texture image
202      * has finished loading.
203      *
204      * @private
205      */
206     this.handleImageTextureLoaded = function (that, texture, image, minFilter, magFilter) {
207         that.gl.bindTexture(that.gl.TEXTURE_2D, texture);        
208         that.gl.texImage2D(that.gl.TEXTURE_2D, 0, that.gl.RGBA, that.gl.RGBA, that.gl.UNSIGNED_BYTE, image);
209         that.gl.texParameteri(that.gl.TEXTURE_2D, that.gl.TEXTURE_MAG_FILTER, magFilter ? magFilter : that.gl.NEAREST);
210         that.gl.texParameteri(that.gl.TEXTURE_2D, that.gl.TEXTURE_MIN_FILTER, minFilter ? minFilter : that.gl.NEAREST);
211         that.gl.texParameteri(that.gl.TEXTURE_2D, that.gl.TEXTURE_WRAP_S, that.gl.CLAMP_TO_EDGE);
212         that.gl.texParameteri(that.gl.TEXTURE_2D, that.gl.TEXTURE_WRAP_T, that.gl.CLAMP_TO_EDGE);
213         if (minFilter == that.gl.NEAREST_MIPMAP_NEAREST
214                 || minFilter == that.gl.LINEAR_MIPMAP_NEAREST
215                     || minFilter == that.gl.NEAREST_MIPMAP_LINEAR
216                     || minFilter == that.gl.LINEAR_MIPMAP_LINEAR) {
217                         that.gl.generateMipmap(that.gl.TEXTURE_2D);
218                     }
219         
220         that.gl.bindTexture(that.gl.TEXTURE_2D, null);      
221     }
222     
223     /**
224      * Transforms a vector to world coordinates.
225      *
226      * @param {vector} vector the vector to transform
227      */
228     this.transformToWorld = function (vector) {
229         var sylvesterVector = $V([vector[0], vector[1], vector[2], 1.0]);
230         var world = this.mvMatrix.matrix ().x (sylvesterVector);
231         return world;
232     }
233     
234     this.transformWorldToScreen = function (world) {
235         if (world.e(3) > 0) {
236             return null;
237         }
238         
239         var screen = this.pMatrix.matrix ().x (world);
240         if (Math.abs (screen.e(4)) < Sylvester.precision) {
241             return null;
242         }
243         var r = {
244             x: (this.gl.viewportWidth / 2) * screen.e(1) / screen.e(4) + this.gl.viewportWidth / 2, 
245             y: - (this.gl.viewportHeight / 2) * screen.e(2) / screen.e(4) + this.gl.viewportHeight / 2
246         };
247         return r;
248     }
249     
250     /**
251      * Transforms a vector to screen coordinates.
252      *
253      * @param {vector} vector the vector to transform
254      * @return the transformed vector, or null if the vector is nearer than the near-z plane.
255      */
256     this.transformToScreen = function (vector) {
257         var world = this.transformToWorld (vector);
258         return this.transformWorldToScreen (world);
259     }
260 };
