 17 /**
 18  * @class Parameters for the adaptive LOD monitor.
 19  */
 20 bigshot.AdaptiveLODMonitorParameters = function (values) {
 21     
 22     this.vrPanorama = null;
 23     
 24     /**
 25      * The target framerate in frames per second. 
 26      * The monitor will try to achieve an average frame render time
 27      * of <i>1 / targetFps</i> seconds.
 28      *
 29      * @default 30
 30      * @type float
 31      */
 32     this.targetFps = 30;
 33     
 34     /**
 35      * The tolerance for the rendering time. The monitor will adjust the
 36      * level of detail if the average frame render time rises above
 37      * <i>target frame render time * (1.0 + tolerance)</i> or falls below
 38      * <i>target frame render time / (1.0 + tolerance)</i>.
 39      *
 40      * @default 0.3
 41      * @type float
 42      */
 43     this.tolerance = 0.3;
 44     
 45     /**
 46      * The rate at which the level of detail is adjusted.
 47      * For detail increase, the detail is multiplied with (1.0 + rate),
 48      * for decrease divided.
 49      */
 50     this.rate = 0.1;
 51     
 52     /**
 53      * Minimum texture magnification.
 54      */
 55     this.minMag = 1.5;
 56     
 57     /**
 58      * Maximum texture magnification.
 59      */
 60     this.maxMag = 16;
 61     
 62     /**
 63      * Texture magnification for HQ render passes.
 64      */
 65     this.hqRenderMag = 1.5;
 66     
 67     /**
 68      * Delay before executing a HQ render pass.
 69      */
 70     this.hqRenderDelay = 2000;
 71     
 72     /**
 73      * Interval for the HQ render pass timer.
 74      */
 75     this.hqRenderInterval = 1000;
 76     
 77     if (values) {
 78         for (var k in values) {
 79             this[k] = values[k];
 80         }
 81     }
 82     
 83     this.merge = function (values, overwrite) {
 84         for (var k in values) {
 85             if (overwrite || !this[k]) {
 86                 this[k] = values[k];
 87             }
 88         }
 89     }
 90     return this;        
 91 };
