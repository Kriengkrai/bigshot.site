 17 /**
 18  * Abstract interface description for a Layer.
 19  *
 20  * @class Abstract interface description for a layer.
 21  * @constructor
 22  */
 23 bigshot.Layer = function () {
 24 }
 25 
 26 bigshot.Layer.prototype = {
 27     /**
 28      * Returns the layer container.
 29      *
 30      * @type HTMLDivElement
 31      */
 32     getContainer : function () {},
 33     
 34     /**
 35      * Sets the maximum number of image tiles that will be visible in the image.
 36      *
 37      * @param {int} x the number of tiles horizontally
 38      * @param {int} y the number of tiles vertically
 39      */
 40     setMaxTiles : function (x, y) {},
 41     
 42     /**
 43      * Called when the image's viewport is resized.
 44      *
 45      * @param {int} w the new width of the viewport, in css pixels
 46      * @param {int} h the new height of the viewport, in css pixels
 47      */
 48     resize : function (w, h) {},
 49     
 50     /**
 51      * Lays out the layer.
 52      *
 53      * @param {number} zoom the zoom level
 54      * @param {number} x0 the x-coordinate of the top-left corner of the top-left tile
 55      * @param {number} y0 the y-coordinate of the top-left corner of the top-left tile
 56      * @param {number} tx0 column number (starting at zero) of the top-left tile
 57      * @param {number} ty0 row number (starting at zero) of the top-left tile
 58      * @param {number} size tileSize
 59      * @param {number} stride offset (vertical and horizontal) from the top-left corner
 60      *                 of a tile to the next tile's top-left corner.
 61      * @param {number} opacity the opacity of the layer.
 62      */
 63     layout : function (zoom, x0, y0, tx0, ty0, size, stride, opacity) {}
 64 };
