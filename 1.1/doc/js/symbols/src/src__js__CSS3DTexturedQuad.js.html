 17 /**
 18  * Creates a textured quad object.
 19  *
 20  * @class An abstraction for textured quads. Used in the
 21  * {@link bigshot.CSS3DTexturedQuadScene}.
 22  *
 23  * @param {point} p the top-left corner of the quad
 24  * @param {vector} u vector pointing from p along the top edge of the quad
 25  * @param {vector} v vector pointing from p along the left edge of the quad
 26  * @param {HTMLImageElement} the image to use.
 27  */
 28 bigshot.CSS3DTexturedQuad = function (p, u, v, image) {
 29     this.p = p;
 30     this.u = u;
 31     this.v = v;
 32     this.image = image;
 33 }
 34 
 35 bigshot.CSS3DTexturedQuad.prototype = {
 36     crossProduct : function (a,b) {
 37         var x = [
 38             a[1]*b[2]-a[2]*b[1], a[2]*b[0]-a[0]*b[2], a[0]*b[1]-a[1]*b[0]
 39         ];
 40         return x;
 41     },
 42     
 43     vecToStr : function (u) {
 44         return (u[0]) + "," + (u[1]) + "," + (u[2]);
 45     },
 46     
 47     quadTransform : function (tl, u, v) {
 48         var w = this.crossProduct (u, v);
 49         var res = 
 50             "matrix3d(" + 
 51             this.vecToStr (u) + ",0," + 
 52         this.vecToStr (v) + ",0," + 
 53         this.vecToStr (w) + ",0," + 
 54         this.vecToStr (tl) + ",1)";
 55         return res;
 56     },
 57     
 58     norm : function (vec) {
 59         return Math.sqrt (vec.x * vec.x + vec.y * vec.y + vec.z * vec.z);
 60     },
 61     
 62     /**
 63      * Renders the quad.
 64      */
 65     render : function (world, scale, view) {
 66         var s = scale / (this.image.width - 1);
 67         var ps = scale * 1.0;
 68         var p = this.p;
 69         var u = this.u;
 70         var v = this.v;
 71         
 72         this.image.style.position = "absolute";
 73         if (!this.image.inWorld || this.image.inWorld != 1) {
 74             world.appendChild (this.image);
 75         }
 76         this.image.inWorld = 2;
 77         this.image.style.WebkitTransformOrigin = "0px 0px 0px";
 78         this.image.style.WebkitTransform = 
 79             this.quadTransform ([(p.x + view.x) * ps, (-p.y + view.y) * ps, (p.z + view.z) * ps], [u.x * s, -u.y * s, u.z * s], [v.x * s, -v.y * s, v.z * s]);
 80     }
 81 }
