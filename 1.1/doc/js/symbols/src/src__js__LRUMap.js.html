 17 /**
 18  * Creates a new, empty, LRUMap instance.
 19  * 
 20  * @class Implementation of a Least-Recently-Used cache map.
 21  * Used by the ImageTileCache to keep track of cache entries.
 22  * @constructor
 23  */
 24 bigshot.LRUMap = function () {
 25     /** 
 26      * Key to last-accessed time mapping.
 27      *
 28      * @type Object
 29      */
 30     this.keyToTime = {};
 31     
 32     /**
 33      * Current time counter. Incremented for each access of
 34      * a key in the map.
 35      * @type int
 36      */
 37     this.counter = 0;
 38     
 39     /** 
 40      * Current size of the map.
 41      * @type int
 42      */
 43     this.size = 0;
 44 }
 45 
 46 bigshot.LRUMap.prototype = {
 47     /**
 48      * Marks access to an item, represented by its key in the map. 
 49      * The key's last-accessed time is updated to the current time
 50      * and the current time is incremented by one step.
 51      *
 52      * @param {String} key the key associated with the accessed item
 53      */
 54     access : function (key) {
 55         this.remove (key);
 56         this.keyToTime[key] = this.counter;
 57         ++this.counter;
 58         ++this.size;
 59     },
 60     
 61     /**
 62      * Removes a key from the map.
 63      *
 64      * @param {String} key the key to remove
 65      * @returns true iff the key existed in the map.
 66      * @type boolean
 67      */
 68     remove : function (key) {
 69         if (this.keyToTime[key]) {
 70             delete this.keyToTime[key];
 71             --this.size;
 72             return true;
 73         } else {
 74             return false;
 75         }
 76     },
 77     
 78     /**
 79      * Returns the current number of keys in the map.
 80      * @type int
 81      */
 82     getSize : function () {
 83         return this.size;
 84     },
 85     
 86     /**
 87      * Returns the key in the map with the lowest
 88      * last-accessed time. This is done as a linear
 89      * search through the map. It could be done much 
 90      * faster with a sorted map, but unless this becomes
 91      * a bottleneck it is just not worth the effort.
 92      * @type String
 93      */
 94     leastUsed : function () {
 95         var least = this.counter + 1;
 96         var leastKey = null;
 97         for (var k in this.keyToTime) {
 98             if (this.keyToTime[k] < least) {
 99                 least = this.keyToTime[k];
100                 leastKey = k;
101             }
102         }
103         return leastKey;
104     }
105 };
