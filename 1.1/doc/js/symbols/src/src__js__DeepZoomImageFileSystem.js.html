 17 /**
 18  * Creates a new instance of a Deep Zoom Image folder-based filesystem adapter.
 19  *
 20  * @augments bigshot.FileSystem
 21  * @class Folder-based filesystem.
 22  * @param {bigshot.ImageParameters or bigshot.VRPanoramaParameters} parameters the associated image parameters
 23  * @constructor
 24  */
 25 bigshot.DeepZoomImageFileSystem = function (parameters) {
 26     this.prefix = "";
 27     this.suffix = "";
 28     
 29     this.DZ_NAMESPACE = "http://schemas.microsoft.com/deepzoom/2009";
 30     this.fullZoomLevel = 0;
 31     this.posterName = "";
 32     this.parameters = parameters;
 33 }
 34 
 35 bigshot.DeepZoomImageFileSystem.prototype = {    
 36     getDescriptor : function () {
 37         var descriptor = {};
 38         
 39         var xml = this.parameters.dataLoader.loadXml (this.parameters.basePath + this.prefix + ".xml", false);
 40         var image = xml.getElementsByTagName ("Image")[0];
 41         var size = xml.getElementsByTagName ("Size")[0];
 42         descriptor.width = parseInt (size.getAttribute ("Width"));
 43         descriptor.height = parseInt (size.getAttribute ("Height"));
 44         descriptor.tileSize = parseInt (image.getAttribute ("TileSize"));
 45         descriptor.overlap = parseInt (image.getAttribute ("Overlap"));
 46         descriptor.suffix = "." + image.getAttribute ("Format")
 47         descriptor.posterSize = descriptor.tileSize;
 48         
 49         this.suffix = descriptor.suffix;
 50         this.fullZoomLevel = Math.ceil (Math.log (Math.max (descriptor.width, descriptor.height)) / Math.LN2);
 51         
 52         descriptor.minZoom = -this.fullZoomLevel;
 53         var posterZoomLevel = Math.ceil (Math.log (descriptor.tileSize) / Math.LN2);
 54         this.posterName = this.getImageFilename (0, 0, posterZoomLevel - this.fullZoomLevel);
 55         return descriptor;
 56     },
 57     
 58     setPrefix : function (prefix) {
 59         this.prefix = prefix;
 60     },
 61     
 62     getPosterFilename : function () {
 63         return this.posterName;
 64     },
 65     
 66     getFilename : function (name) {
 67         return this.parameters.basePath + this.prefix + "/" + name;
 68     },
 69     
 70     getImageFilename : function (tileX, tileY, zoomLevel) {
 71         var dziZoomLevel = this.fullZoomLevel + zoomLevel;
 72         var key = dziZoomLevel + "/" + tileX + "_" + tileY + this.suffix;
 73         return this.getFilename (key);
 74     }
 75 };
