 17 /**
 18  * @class A utility class for making a document element "full screen", or as close to that
 19  * as browser security allows.
 20  */
 21 bigshot.FullScreen = function (container) {
 22     this.browser = new bigshot.Browser ();
 23     
 24     this.container = container;
 25     
 26     this.isFullScreen = false;
 27     this.savedBodyStyle = null;
 28     this.savedParent = null;
 29     this.savedSize = null;
 30     this.expanderDiv = null;
 31     this.restoreSize = false;
 32     
 33     this.onCloseHandlers = new Array ();
 34     this.onResizeHandlers = new Array ();
 35     
 36     this.init = function () {
 37         
 38     };
 39     
 40     this.getRootElement = function () {
 41         return this.div;
 42     };
 43     
 44     this.addOnClose = function (onClose) {
 45         this.onCloseHandlers.push (onClose);
 46     };
 47     
 48     this.onClose = function () {
 49         for (var i = 0; i < this.onCloseHandlers.length; ++i) {
 50             this.onCloseHandlers[i] ();
 51         }
 52     };
 53     
 54     this.addOnResize = function (onResize) {
 55         this.onResizeHandlers.push (onResize);
 56     };
 57     
 58     this.onResize = function () {
 59         for (var i = 0; i < this.onResizeHandlers.length; ++i) {
 60             this.onResizeHandlers[i] ();
 61         }
 62     };
 63     
 64     this.open = function () {
 65         this.isFullScreen = true;
 66         
 67         this.savedParent = this.container.parentNode;
 68         this.savedSize = {
 69             width : this.container.style.width,
 70             height : this.container.style.height
 71         };
 72         this.savedBodyStyle = document.body.style.cssText;
 73         
 74         document.body.style.overflow = "hidden";
 75         
 76         this.expanderDiv = document.createElement ("div");
 77         this.expanderDiv.style.position = "absolute";
 78         this.expanderDiv.style.top = "0px";
 79         this.expanderDiv.style.left = "0px";
 80         this.expanderDiv.style.width = Math.max (window.innerWidth, document.documentElement.clientWidth) + "px";
 81         this.expanderDiv.style.height = Math.max (window.innerHeight, document.documentElement.clientHeight) + "px";
 82         
 83         document.body.appendChild (this.expanderDiv);
 84         
 85         this.div = document.createElement ("div");
 86         this.div.style.position = "fixed";
 87         this.div.style.top = window.pageYOffset + "px";
 88         this.div.style.left = window.pageXOffset + "px";
 89         
 90         this.div.style.width = window.innerWidth + "px";
 91         this.div.style.height = window.innerHeight + "px";
 92         this.div.style.zIndex = "9998";
 93         
 94         this.div.appendChild (this.container);
 95         
 96         //this.container.style.width = window.innerWidth + "px";
 97         //this.container.style.height = window.innerHeight + "px";
 98 
 99         document.body.appendChild (this.div);
100         
101         var that = this;
102         var resizeHandler = function (e) {
103             setTimeout (function () {
104                     that.div.style.width = window.innerWidth + "px";
105                     that.div.style.height = window.innerHeight + "px";                    
106                     setTimeout (function () {
107                             that.onResize ();
108                         }, 1);
109                 }, 1);
110         };
111         
112         
113         var rotationHandler = function (e) {
114             that.expanderDiv.style.width = Math.max (window.innerWidth, document.documentElement.clientWidth) + "px";
115             that.expanderDiv.style.height = Math.max (window.innerHeight, document.documentElement.clientHeight) + "px";
116             setTimeout (function () {
117                     that.div.style.top = window.pageYOffset + "px";
118                     that.div.style.left = window.pageXOffset + "px";
119                     that.div.style.width = window.innerWidth + "px";
120                     that.div.style.height = window.innerHeight + "px";
121                     setTimeout (function () {
122                             that.onResize ();
123                         }, 1);
124                 }, 1);
125         };
126         
127         this.exitFullScreenHandler = function () {
128             that.isFullScreen = false;
129             that.browser.unregisterListener (document, "keydown", escHandler);
130             that.browser.unregisterListener (window, "resize", resizeHandler);
131             that.browser.unregisterListener (document.body, "orientationchange", rotationHandler);
132             if (that.restoreSize) {
133                 that.container.style.width = that.savedSize.width;
134                 that.container.style.height = that.savedSize.height;
135             }     
136             
137             document.body.style.cssText = that.savedBodyStyle;
138             
139             that.savedParent.appendChild (that.container);
140             document.body.removeChild (that.div);
141             document.body.removeChild (that.expanderDiv);
142             
143             that.onResize ();            
144             that.onClose ();
145             setTimeout (function () {
146                     that.onResize ();
147                 }, 1);
148         };
149         
150         var escHandler = function (e) {
151             if (e.keyCode == 27) {
152                 that.exitFullScreenHandler ();
153             }
154         };
155         this.browser.registerListener (document, "keydown", escHandler, false);
156         this.browser.registerListener (window, "resize", resizeHandler, false);
157         this.browser.registerListener (document.body, "orientationchange", rotationHandler, false);
158         
159         this.onResize ();
160         
161         return this.exitFullScreenHandler;
162     };
163     
164     this.close = function () {
165         this.exitFullScreenHandler ();
166     };
167     
168     this.init ();
169     return this;
170 };
