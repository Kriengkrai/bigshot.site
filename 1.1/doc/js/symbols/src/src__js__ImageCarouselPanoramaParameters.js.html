 17 /**
 18  * Creates a new set of panorama parameters.
 19  *
 20  * @class Parameters for an {@link bigshot.ImageCarouselPanorama}
 21  */
 22 bigshot.ImageCarouselPanoramaParameters = function (values) {
 23     /**
 24      * Suffix to append to the tile filenames. Typically <code>".jpg"</code> or 
 25      * <code>".png"</code>.
 26      *
 27      * @default <i>Optional</i> set by MakeImagePyramid and loaded from descriptor
 28      * @type String
 29      */
 30     this.suffix = null;
 31     
 32     /**
 33      * For {@link bigshot.ImageCarouselPanorama}, the {@code div} to render into.
 34      *
 35      * @type HTMLDivElement
 36      */
 37     this.container = null;
 38     
 39     /**
 40      * Base path for the image. This is filesystem dependent; but for the two most common cases
 41      * the following should be set
 42      *
 43      * <ul>
 44      * <li><b>archive</b>= The basePath is <code>"<path>/bigshot.php?file=<path-to-bigshot-archive-relative-to-bigshot.php>"</code>;
 45      *     for example; <code>"/bigshot.php?file=images/bigshot-sample.bigshot"</code>.
 46      * <li><b>folder</b>= The basePath is <code>"<path-to-image-folder>"</code>;
 47      *     for example; <code>"/images/bigshot-sample"</code>.
 48      * </ul>
 49      *
 50      * @type String
 51      */
 52     this.basePath = null;
 53     
 54     /**
 55      * The file system type. Used to create a filesystem instance unless
 56      * the fileSystem field is set. Possible values are <code>"archive"</code> or 
 57      * <code>"folder"</code>
 58      *
 59      * @type String
 60      * @default "folder"
 61      */
 62     this.fileSystemType = "folder";
 63     
 64     /**
 65      * A reference to a filesystem implementation. If set; it overrides the
 66      * fileSystemType field.
 67      *
 68      * @default set depending on value of bigshot.VRPanoramaParameters#fileSystemType
 69      * @type bigshot.FileSystem
 70      */
 71     this.fileSystem = null;
 72     
 73     this.steps = 0;
 74     
 75     if (values) {
 76         for (var k in values) {
 77             this[k] = values[k];
 78         }
 79     }
 80     
 81     this.merge = function (values, overwrite) {
 82         for (var k in values) {
 83             if (overwrite || !this[k]) {
 84                 this[k] = values[k];
 85             }
 86         }
 87     }
 88     return this;        
 89 }
