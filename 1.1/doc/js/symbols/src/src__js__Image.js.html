 18 /**
 19  * Creates a new tiled image viewer. (Note: See {@link bigshot.Image#dispose} for important information.)
 20  *
 21  * @example
 22  * var bsi = new bigshot.Image (
 23  *     new bigshot.ImageParameters ({
 24  *         basePath : "/bigshot.php?file=myshot.bigshot",
 25  *         fileSystemType : "archive",
 26  *         container : document.getElementById ("bigshot_div")
 27  *     }));
 28  *
 29  * @param {bigshot.ImageParameters} parameters the image parameters. Required fields are: <code>basePath</code> and <code>container</code>.
 30  * If you intend to use the archive filesystem, you need to set the <code>fileSystemType</code> to <code>"archive"</code>
 31  * as well.
 32  * @see bigshot.Image#dispose
 33  * @class A tiled, zoomable image viewer.
 34  * @constructor
 35  */     
 36 bigshot.Image = function (parameters) {
 37     bigshot.setupFileSystem (parameters);
 38     parameters.merge (parameters.fileSystem.getDescriptor (), false);
 39     
 40     this.parameters = parameters;
 41     this.flying = 0;
 42     this.container = parameters.container;
 43     this.x = parameters.width / 2.0;
 44     this.y = parameters.height / 2.0;
 45     this.zoom = 0.0;
 46     this.width = parameters.width;
 47     this.height = parameters.height;
 48     this.minZoom = parameters.minZoom;
 49     this.maxZoom = 2.0;
 50     this.tileSize = parameters.tileSize;
 51     this.overlap = 0;
 52     this.imageTileCache = null;
 53     
 54     this.dragStart = null;
 55     this.dragged = false;
 56     
 57     this.layers = new Array ();
 58     
 59     this.fullScreenHandler = null;
 60     this.currentGesture = null;
 61     
 62     var that = this;
 63     this.onresizeHandler = function (e) {
 64         that.onresize ();
 65     }
 66     
 67     /**
 68      * Helper function to consume events.
 69      * @private
 70      */
 71     var consumeEvent = function (event) {
 72         if (event.preventDefault) {
 73             event.preventDefault ();
 74         }
 75         return false;
 76     };
 77     
 78     /**
 79      * Helper function to translate touch events to mouse-like events.
 80      * @private
 81      */
 82     var translateEvent = function (event) {
 83         if (event.clientX) {
 84             return event;
 85         } else {
 86             return {
 87                 clientX : event.changedTouches[0].clientX,
 88                 clientY : event.changedTouches[0].clientY
 89             };
 90         };
 91     };
 92     
 93     this.thisTileCache = new bigshot.ImageTileCache (function () {
 94             that.layout ();     
 95         }, null, parameters);
 96     
 97     this.addLayer (
 98         new bigshot.TileLayer (this, parameters, 0, 0, this.thisTileCache)
 99     );
100     this.resize ();
101     
102     this.browser.registerListener (parameters.container, "DOMMouseScroll", function (e) {
103             that.mouseWheel (e);
104             return consumeEvent (e);
105         }, false);
106     this.browser.registerListener (parameters.container, "mousewheel", function (e) {
107             that.mouseWheel (e);
108             return consumeEvent (e);
109         }, false);
110     this.browser.registerListener (parameters.container, "dblclick", function (e) {
111             that.mouseDoubleClick (e);
112             return consumeEvent (e);
113         }, false);
114     this.browser.registerListener (parameters.container, "mousedown", function (e) {
115             that.dragMouseDown (e);
116             return consumeEvent (e);
117         }, false);
118     this.browser.registerListener (parameters.container, "gesturestart", function (e) {
119             that.gestureStart (e);
120             return consumeEvent (e);
121         }, false);
122     this.browser.registerListener (parameters.container, "gesturechange", function (e) {
123             that.gestureChange (e);
124             return consumeEvent (e);
125         }, false);
126     this.browser.registerListener (parameters.container, "gestureend", function (e) {
127             that.gestureEnd (e);
128             return consumeEvent (e);
129         }, false);
130     this.browser.registerListener (parameters.container, "touchstart", function (e) {
131             that.dragMouseDown (translateEvent (e));
132             return consumeEvent (e);
133         }, false);
134     this.browser.registerListener (parameters.container, "mouseup", function (e) {
135             that.dragMouseUp (e);
136             return consumeEvent (e);
137         }, false);
138     this.browser.registerListener (parameters.container, "touchend", function (e) {
139             that.dragMouseUp (translateEvent (e));
140             return consumeEvent (e);
141         }, false);
142     this.browser.registerListener (parameters.container, 'mousemove', function (e) {
143             that.dragMouseMove (e);
144             return consumeEvent (e);
145         }, false);
146     this.browser.registerListener (parameters.container, 'mouseout', function (e) {
147             //that.dragMouseUp (e);
148             return consumeEvent (e);
149         }, false);
150     this.browser.registerListener (parameters.container, 'touchmove', function (e) {
151             that.dragMouseMove (translateEvent (e));
152             return consumeEvent (e);
153         }, false);
154     this.browser.registerListener (window, 'resize', that.onresizeHandler, false);
155     this.zoomToFit ();
156     return this;
157 }

162     /**
163      * Lays out all layers according to the current 
164      * x, y and zoom values.
165      *
166      * @public
167      */
168     layout : function () {
169         var zoomLevel = Math.min (0, Math.ceil (this.zoom));
170         var zoomFactor = Math.pow (2, zoomLevel);
171         var tileWidthInRealPixels = this.tileSize / zoomFactor;
172         
173         var fractionalZoomFactor = Math.pow (2, this.zoom - zoomLevel);
174         var tileDisplayWidth = this.tileSize * fractionalZoomFactor;
175         
176         var widthInTiles = this.width / tileWidthInRealPixels;
177         var heightInTiles = this.height / tileWidthInRealPixels;
178         var centerInTilesX = this.x / tileWidthInRealPixels;
179         var centerInTilesY = this.y / tileWidthInRealPixels;
180         
181         var viewportWidth = this.container.clientWidth;
182         var viewportHeight = this.container.clientHeight;
183         
184         var topLeftInTilesX = centerInTilesX - (viewportWidth / 2) / tileDisplayWidth;
185         var topLeftInTilesY = centerInTilesY - (viewportHeight / 2) / tileDisplayWidth;
186         
187         var topLeftTileX = Math.floor (topLeftInTilesX);
188         var topLeftTileY = Math.floor (topLeftInTilesY);
189         var topLeftTileXoffset = Math.round ((topLeftInTilesX - topLeftTileX) * tileDisplayWidth);
190         var topLeftTileYoffset = Math.round ((topLeftInTilesY - topLeftTileY) * tileDisplayWidth);
191         
192         for (var i = 0; i < this.layers.length; ++i) {
193             this.layers[i].layout (
194                 this.zoom, 
195                 -topLeftTileXoffset - tileDisplayWidth, -topLeftTileYoffset - tileDisplayWidth, 
196                 topLeftTileX - 1, topLeftTileY - 1, 
197                 Math.ceil (tileDisplayWidth), Math.ceil (tileDisplayWidth), 
198                 1.0);
199         }
200     },

202     /**
203      * Resizes the layers of this image.
204      *
205      * @public

208         var tilesW = Math.ceil (2 * this.container.clientWidth / this.tileSize) + 2;

210         for (var i = 0; i < this.layers.length; ++i) {
211             this.layers[i].resize (tilesW, tilesH);
212         }

252     /**
253      * Sets the current zoom value.
254      *
255      * @private
256      * @param {number} zoom the zoom value.
257      * @param {boolean} layout trigger a viewport update after setting. Defaults to <code>true</code>.
258      */
259     setZoom : function (zoom, updateViewport) {
260         this.zoom = Math.min (this.maxZoom, Math.max (zoom, this.minZoom));
261         var zoomLevel = Math.ceil (this.zoom);
262         var zoomFactor = Math.pow (2, zoomLevel);
263         var maxTileX = Math.ceil (zoomFactor * this.width / this.tileSize);
264         var maxTileY = Math.ceil (zoomFactor * this.height / this.tileSize);
265         for (var i = 0; i < this.layers.length; ++i) {
266             this.layers[i].setMaxTiles (maxTileX, maxTileY);
267         }
268     },

355     dragMouseDown : function (event) {
356         this.dragStart = {
357             x : event.clientX,
358             y : event.clientY
359         };
360         this.dragged = false;
361     },
362     
363     /**
364      * Handles a mouse drag event by panning the image.
365      * Also sets the dragged flag to indicate that the
366      * following <code>click</code> event should be ignored.
367      * @private
368      */
369     dragMouseMove : function (event) {
370         if (this.currentGesture == null && this.dragStart != null) {
371             var delta = {
372                 x : event.clientX - this.dragStart.x,
373                 y : event.clientY - this.dragStart.y
374             };
375             if (delta.x != 0 || delta.y != 0) {
376                 this.dragged = true;
377             }
378             var zoomFactor = Math.pow (2, this.zoom);
379             var realX = delta.x / zoomFactor;
380             var realY = delta.y / zoomFactor;
381             this.moveTo (this.x - realX, this.y - realY);
382             this.dragStart = {
383                 x : event.clientX,
384                 y : event.clientY
385             };
386         }
387     },
388     
389     /**
390      * Ends a drag event by freeing the associated structures.
391      * @private
392      */
393     dragMouseUp : function (event) {
394         if (this.dragStart != null) {
395             this.dragStart = null;
396             if (!this.dragged && this.parameters.touchUI) {
397                 this.mouseClick (event);
398             }
399         }
400     },

402     /**
403      * Mouse double-click handler. Pans to the clicked point and
404      * zooms in half a zoom level (approx 40%).
405      * @private
406      */
407     mouseDoubleClick : function (event) {
408         var elementPos = this.browser.getElementPosition (this.container);
409         var clickPos = {
410             x : event.clientX - elementPos.x - this.container.clientWidth / 2,
411             y : event.clientY - elementPos.y - this.container.clientHeight / 2
412         };
413         var scale = Math.pow (2, this.zoom);
414         clickPos.x /= scale;
415         clickPos.y /= scale;
416         this.flyTo (this.x + clickPos.x, this.y + clickPos.y, this.zoom + 0.5);
417     },

429     /**
430      * Stops any current flyTo operation and sets the current position.
431      *
432      * @param [x] the new x-coordinate
433      * @param [y] the new y-coordinate
434      * @param [zoom] the new zoom level
435      * @param [updateViewport] if the viewport should be updated, defaults to <code>true</code>
436      * @public
437      */
438     moveTo : function (x, y, zoom) {
439         this.stopFlying ();
440         if (x != null || y != null) {
441             this.setPosition (x, y, false);
442         }
443         if (zoom != null) {
444             this.setZoom (zoom, false);
445         }
446         this.layout ();
447     },
448     
449     /**
450      * Sets the current position.
451      *
452      * @param [x] the new x-coordinate
453      * @param [y] the new y-coordinate
454      * @param [updateViewport] if the viewport should be updated, defaults to <code>true</code>
455      * @private
456      */
457     setPosition : function (x, y, updateViewport) {
458         if (x != null) {
459             if (this.parameters.wrapX) {
460                 if (x < 0 || x >= this.width) {
461                     x = (x + this.width) % this.width;
462                 }
463             }
464             this.x = Math.max (0, Math.min (this.width, x));
465         }
466         
467         if (y != null) {
468             if (this.parameters.wrapY) {
469                 if (y < 0 || y >= this.height) {
470                     y = (y + this.height) % this.height;
471                 }
472             }
473             this.y = Math.max (0, Math.min (this.height, y));
474         }
475         
476         if (updateViewport != false) {
477             this.layout ();
478         }
479     },

481     /**
482      * Helper function for calculating zoom levels.
483      *
484      * @public
485      * @returns the zoom level at which the given number of full-image pixels

488      * @param {number} containerDimension the container dimension in screen pixels
489      */

491         var scale = containerDimension / imageDimension;
492         return Math.log (scale) / Math.LN2;

559     /**
560      * Handles mouse wheel actions.
561      * @private
562      */
563     mouseWheelHandler : function (delta) {
564         if (delta > 0) {
565             this.flyTo (this.x, this.y, this.getZoom () + 0.5);
566         } else if (delta < 0) {
567             this.flyTo (this.x, this.y, this.getZoom () - 0.5);

571     /**
572      * Translates mouse wheel events.
573      * @private
574      */
575     mouseWheel : function (event){
576         var delta = 0;
577         if (!event) /* For IE. */
578             event = window.event;
579         if (event.wheelDelta) { /* IE/Opera. */
580             delta = event.wheelDelta / 120;
581             /*
582              * In Opera 9, delta differs in sign as compared to IE.
583              */
584             if (window.opera)
585                 delta = -delta;
586         } else if (event.detail) { /* Mozilla case. */
587             /*
588              * In Mozilla, sign of delta is different than in IE.
589              * Also, delta is multiple of 3.
590              */
591             delta = -event.detail;
592         }
593         
594         /*
595          * If delta is nonzero, handle it.
596          * Basically, delta is now positive if wheel was scrolled up,
597          * and negative, if wheel was scrolled down.
598          */
599         if (delta) {
600             this.mouseWheelHandler (delta);
601         }
602         
603         /*
604          * Prevent default actions caused by mouse wheel.
605          * That might be ugly, but we handle scrolls somehow
606          * anyway, so don't bother here..
607          */
608         if (event.preventDefault) {
609             event.preventDefault ();
610         }
611         event.returnValue = false;
612     },

652     /**
653      * Smoothly flies to the specified position.
654      *
655      * @public
656      * @param {number} [x] the new x-coordinate
657      * @param {number} [y] the new y-coordinate
658      * @param {number} [zoom] the new zoom level
659      */
660     flyTo : function (x, y, zoom) {
661         var that = this;
662         
663         x = x != null ? x : this.x;
664         y = y != null ? y : this.y;
665         zoom = zoom != null ? zoom : this.zoom;
666         
667         var targetX = Math.max (0, Math.min (this.width, x));
668         var targetY = Math.max (0, Math.min (this.height, y));
669         
670         var targetZoom = Math.min (this.maxZoom, Math.max (zoom, this.minZoom));
671         
672         this.flying++;
673         var flyingAtStart = this.flying;
674         
675         var approach = function (current, target, step) {
676             return current + (target - current) * step;
677         };
678         
679         var iter = function () {
680             if (that.flying == flyingAtStart) {
681                 var nx = approach (that.x, targetX, 0.5);
682                 var ny = approach (that.y, targetY, 0.5);
683                 var nz = approach (that.zoom, targetZoom, 0.5);
684                 var done = true;
685                 if (Math.abs (that.x - targetX) < 1.0) {
686                     nx = targetX;
687                 } else {
688                     done = false;
689                 }
690                 if (Math.abs (that.y - targetY) < 1.0) {
691                     ny = targetY;
692                 } else {
693                     done = false;
694                 }
695                 if (Math.abs (that.zoom - targetZoom) < 0.02) {
696                     nz = targetZoom;
697                 } else {
698                     done = false;
699                 }
700                 that.setPosition (nx, ny, false);
701                 that.setZoom (nz, false);
702                 that.layout ();
703                 if (!done) {
704                     setTimeout (iter, 20);
705                 }
706             };
707         }
708         setTimeout (iter, 20);
709     },

711     /**
712      * Returns the maximum zoom level at which a rectangle with the given dimensions
713      * fit into the viewport.
714      *
715      * @public
716      * @param {number} w the width of the rectangle, given in full-image pixels
717      * @param {number} h the height of the rectangle, given in full-image pixels
718      */        
719     rectVisibleAtZoomLevel : function (w, h) {
720         return Math.min (
721             this.fitZoom (w, this.container.clientWidth),
722             this.fitZoom (h, this.container.clientHeight));
723     },
724     
725     /**
726      * Returns the base size in screen pixels of the two zoom touch areas.
727      * The zoom out border will be getTouchAreaBaseSize() pixels wide,
728      * and the center zoom in hotspot will be 2*getTouchAreaBaseSize() pixels wide
729      * and tall.
730      * @type number
731      * @public
732      */
733     getTouchAreaBaseSize : function () {
734         var averageSize = ((this.container.clientWidth + this.container.clientHeight) / 2) * 0.2;
735         return Math.min (averageSize, Math.min (this.container.clientWidth, this.container.clientHeight) / 6);
736     },

738     /**
739      * Handles mouse click events. If the touch UI is active,
740      * we'll pan and/or zoom, as appropriate. If not, we just ignore
741      * the event.
742      * @private
743      */
744     mouseClick : function (event) {
745         if (!this.parameters.touchUI) {
746             return;
747         }
748         if (this.dragged) {
749             return;
750         }
751         var elementPos = this.browser.getElementPosition (this.container);
752         var clickPos = {
753             x : event.clientX - elementPos.x - this.container.clientWidth / 2,
754             y : event.clientY - elementPos.y - this.container.clientHeight / 2
755         };
756         
757         var zoomOutBorderSize = this.getTouchAreaBaseSize ();
758         var zoomInHotspotSize = this.getTouchAreaBaseSize ();
759         
760         if (Math.abs (clickPos.x) > (this.container.clientWidth / 2 - zoomOutBorderSize) || Math.abs (clickPos.y) > (this.container.clientHeight / 2 - zoomOutBorderSize)) {
761             this.flyTo (this.x, this.y, this.zoom - 0.5);
762         } else {
763             var newZoom = this.zoom;
764             if (Math.abs (clickPos.x) < zoomInHotspotSize && Math.abs (clickPos.y) < zoomInHotspotSize) {
765                 newZoom += 0.5;
766             }
767             var scale = Math.pow (2, this.zoom);
768             clickPos.x /= scale;
769             clickPos.y /= scale;
770             this.flyTo (this.x + clickPos.x, this.y + clickPos.y, newZoom);
771         }
772     },

774     /**
775      * Briefly shows the touch ui zones. See the {@link bigshot.ImageParameters#touchUI}
776      * documentation for an explanation of the touch ui.
777      * 
778      * @public
779      * @see bigshot.ImageParameters#touchUI
780      * @param {int} [delay] milliseconds before fading out
781      * @param {int} [fadeOut] milliseconds to fade out the zone overlays in
782      */
783     showTouchUI : function (delay, fadeOut) {
784         if (!delay) {
785             delay = 2500;
786         }
787         if (!fadeOut) {
788             fadeOut = 1000;
789         }
790         
791         var zoomOutBorderSize = this.getTouchAreaBaseSize ();
792         var zoomInHotspotSize = this.getTouchAreaBaseSize ();
793         var centerX = this.container.clientWidth / 2;

796         var frameDiv = document.createElement ("div");

807         centerSpot.style.position = "relative";
808         centerSpot.style.background = "black";
809         centerSpot.style.textAlign = "center";
810         centerSpot.style.top = (centerY - zoomInHotspotSize) + "px";
811         centerSpot.style.left = (centerX - zoomInHotspotSize) + "px";
812         centerSpot.style.width = (2 * zoomInHotspotSize) + "px";
813         centerSpot.style.height = (2 * zoomInHotspotSize) + "px";
814         
815         frameDiv.appendChild (centerSpotAnchor);
816         centerSpotAnchor.appendChild (centerSpot);
817         centerSpot.innerHTML = "<span style='display:inline-box; position:relative; vertical-align:middle; font-size: 20pt; top: 10pt; color:white'>ZOOM IN</span>";
818         
819         var zoomOutBorderAnchor = document.createElement ("div");
820         zoomOutBorderAnchor.style.position = "absolute";
821         
822         var zoomOutBorder = document.createElement ("div");
823         zoomOutBorder.style.position = "relative";
824         zoomOutBorder.style.border = zoomOutBorderSize + "px solid black";
825         zoomOutBorder.style.top = "0px";
826         zoomOutBorder.style.left = "0px";
827         zoomOutBorder.style.textAlign = "center";
828         zoomOutBorder.style.width = this.container.clientWidth + "px";
829         zoomOutBorder.style.height = this.container.clientHeight + "px";
830         zoomOutBorder.style.MozBoxSizing = 
831             zoomOutBorder.style.boxSizing = 
832             zoomOutBorder.style.WebkitBoxSizing = 
833             "border-box";
834         
835         zoomOutBorder.innerHTML = "<span style='position:relative; font-size: 20pt; top: -25pt; color:white'>ZOOM OUT</span>";
836         
837         zoomOutBorderAnchor.appendChild (zoomOutBorder);
838         frameDiv.appendChild (zoomOutBorderAnchor);
839         
840         this.container.appendChild (frameDiv);
841         
842         var that = this;
843         var opacity = 0.9;
844         var fadeOutSteps = fadeOut / 50;
845         if (fadeOutSteps < 1) {
846             fadeOutSteps = 1;
847         }
848         var iter = function () {
849             opacity = opacity - (0.9 / fadeOutSteps);
850             if (opacity < 0.0) {
851                 that.container.removeChild (frameDiv);
852             } else {
853                 frameDiv.style.opacity = opacity;
854                 setTimeout (iter, 50);
855             }
856         };
857         setTimeout (iter, delay);

872     /**
873      * Maximizes the image to cover the browser viewport.
874      * The container div is removed from its parent node upon entering 
875      * full screen mode. When leaving full screen mode, the container
876      * is appended to its old parent node. To avoid rearranging the
877      * nodes, wrap the container in an extra div.
878      *
879      * <p>For unknown reasons (probably security), browsers will
880      * not let you open a window that covers the entire screen.
881      * Even when specifying "fullscreen=yes", all you get is a window
882      * that has a title bar and only covers the desktop (not any task
883      * bars or the like). For now, this is the best that I can do,
884      * but should the situation change I'll update this to be
885      * full-screen<i>-ier</i>.
886      * @public
887      */
888     fullScreen : function (onClose) {
889         if (this.fullScreenHandler) {
890             return;
891         }
892         
893         var message = document.createElement ("div");
894         message.style.position = "absolute";
895         message.style.fontSize = "16pt";
896         message.style.top = "128px";
897         message.style.width = "100%";
898         message.style.color = "white";
899         message.style.padding = "16px";
900         message.style.zIndex = "9999";
901         message.style.textAlign = "center";
902         message.style.opacity = "0.75";
903         message.innerHTML = "<span style='border-radius: 16px; -moz-border-radius: 16px; padding: 16px; padding-left: 32px; padding-right: 32px; background:black'>Press Esc to exit full screen mode.</span>";
904         
905         var that = this;
906         
907         this.fullScreenHandler = new bigshot.FullScreen (this.container);
908         this.fullScreenHandler.restoreSize = true;
909         
910         this.fullScreenHandler.addOnResize (function () {
911                 if (that.fullScreenHandler && that.fullScreenHandler.isFullScreen) {
912                     that.container.style.width = window.innerWidth + "px";
913                     that.container.style.height = window.innerHeight + "px";                
914                 }
915                 that.onresize ();
916             });
917         
918         this.fullScreenHandler.addOnClose (function () {
919                 if (message.parentNode) {
920                     try {
921                         div.removeChild (message);
922                     } catch (x) {
923                     }
924                 }
925                 that.fullScreenHandler = null;
926             });
927         
928         if (onClose) {
929             this.fullScreenHandler.addOnClose (function () {
930                     onClose ();
931                 });
932         }
933         
934         this.fullScreenHandler.open ();
935         this.fullScreenHandler.getRootElement ().appendChild (message);
936         
937         setTimeout (function () {
938                 var opacity = 0.75;
939                 var iter = function () {
940                     opacity -= 0.02;
941                     if (message.parentNode) {
942                         if (opacity <= 0) {
943                             try {
944                                 div.removeChild (message);
945                             } catch (x) {}
946                         } else {
947                             message.style.opacity = opacity;
948                             setTimeout (iter, 20);
949                         }
950                     }
951                 };
952                 setTimeout (iter, 20);
953             }, 3500);
954         
955         return function () {
956             that.fullScreenHandler.close ();
957         };        
958     },
