 17 /**
 18  * @class Object-oriented support functions, used to make JavaScript
 19  * a bit more palatable to a Java-head.
 20  */
 21 bigshot.object = {
 22     /**
 23      *
 24      * @param {Function} base the base-class
 25      * @param {Function} derived the derived-class
 26      */
 27     extend : function (base, derived) {
 28         for (var k in derived.prototype) {
 29             if (base.prototype[k]) {
 30                 base.prototype[k]._super = derived.prototype[k];
 31             } else {
 32                 base.prototype[k] = derived.prototype[k];
 33             }
 34         }
 35     },
 36     
 37     /**
 38      * Utility function to show an object's fields in a message box.
 39      *
 40      * @param {Object} o the object
 41      */
 42     alertr : function (o) {
 43         var sb = "";
 44         for (var k in o) {
 45             sb += k + ":" + o[k] + "\n";
 46         }
 47         alert (sb);
 48     },
 49     
 50     /**
 51      * Utility function to show an object's fields in the console log.
 52      *
 53      * @param {Object} o the object
 54      */
 55     logr : function (o) {
 56         var sb = "";
 57         for (var k in o) {
 58             sb += k + ":" + o[k] + "\n";
 59         }
 60         if (console) {
 61             console.log (sb);
 62         }
 63     }
 64 };
