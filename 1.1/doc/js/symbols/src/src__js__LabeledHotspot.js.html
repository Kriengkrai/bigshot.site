 17 /**
 18  * Creates a new labeled hotspot instance.
 19  *
 20  * @class A hotspot with a label under it. The label element can be accessed using
 21  * the getLabel method and styled as any HTMLElement. See {@link bigshot.HotspotLayer} for 
 22  * examples.
 23  *
 24  * @see bigshot.HotspotLayer
 25  * @param {number} x x-coordinate of the top-left corner, given in full image pixels
 26  * @param {number} y y-coordinate of the top-left corner, given in full image pixels
 27  * @param {number} w width of the hotspot, given in full image pixels
 28  * @param {number} h height of the hotspot, given in full image pixels
 29  * @param {String} labelText text of the label
 30  * @augments bigshot.Hotspot
 31  * @constructor
 32  */
 33 bigshot.LabeledHotspot = function (x, y, w, h, labelText) {
 34     var hs = new bigshot.Hotspot (x, y, w, h);
 35     
 36     /**
 37      * Returns the label element.
 38      *
 39      * @type HTMLDivElement
 40      */
 41     this.getLabel = function () {
 42         return this.label;
 43     };
 44     
 45     this.layout = function (x0, y0, zoomFactor) {
 46         this._super.layout (x0, y0, zoomFactor);
 47         var labelSize = this.browser.getElementSize (this.label);
 48         var sw = this.w * zoomFactor;
 49         var sh = this.h * zoomFactor;
 50         this.label.style.top = (sh + 4) + "px";
 51         this.label.style.left = ((sw - labelSize.w) / 2) + "px";
 52     };
 53     
 54     this.label = document.createElement ("div");
 55     this.label.style.position = "relative";
 56     this.label.style.display = "inline-block";
 57     
 58     hs.getElement ().appendChild (this.label);
 59     this.label.innerHTML = labelText;
 60     
 61     return bigshot.object.extend (hs, this);
 62 };
