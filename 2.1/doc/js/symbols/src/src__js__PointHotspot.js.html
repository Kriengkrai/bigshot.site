 17 /**
 18  * Creates a new labeled hotspot instance.
 19  *
 20  * @class A point hotspot consisting of an image.
 21  *
 22  * @see bigshot​.HotspotLayer
 23  * @param {number} x x-coordinate of the center corner, given in full image pixels
 24  * @param {number} y y-coordinate of the center corner, given in full image pixels
 25  * @param {number} w width of the hotspot, given in screen pixels
 26  * @param {number} h height of the hotspot, given in screen pixels
 27  * @param {number} xo x-offset, given in screen pixels
 28  * @param {number} yo y-offset, given in screen pixels
 29  * @param {HTMLElement} element the HTML element to position
 30  * @param {String} [imageUrl] the image to use as hotspot sprite
 31  * @augments bigshot​.Hotspot
 32  */
 33 bigshot​.PointHotspot = function (x, y, w, h, xo, yo, imageUrl) {
 34     bigshot​.Hotspot​.call (this, x, y, w, h);
 35     this​.xo = xo;
 36     this​.yo = yo;
 37     
 38     if (imageUrl) {
 39         var el = this​.getElement ();
 40         el​.style​.backgroundImage = "url('" + imageUrl + "')";
 41         el​.style​.backgroundRepeat = "no-repeat";
 42     }
 43 }
 44 
 45 bigshot​.PointHotspot​.prototype = {
 46     /**
 47      * Returns the label element.
 48      *
 49      * @type HTMLDivElement
 50      */
 51     getLabel : function () {
 52         return this​.label;
 53     },
 54     
 55     layout : function (x0, y0, zoomFactor) {
 56         var sx = this​.x * zoomFactor + x0 + this​.xo;
 57         var sy = this​.y * zoomFactor + y0 + this​.yo;
 58         this​.element​.style​.top = sy + "px";
 59         this​.element​.style​.left = sx + "px";
 60         this​.element​.style​.width = this​.w + "px";
 61         this​.element​.style​.height = this​.h + "px";
 62     }
 63 };
 64 
 65 bigshot​.Object​.extend (bigshot​.PointHotspot, bigshot​.Hotspot);
