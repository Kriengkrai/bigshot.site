 17 /**
 18  * Abstract base class for panorama hotspots.
 19  *
 20  * @class Abstract base class for panorama hotspots.
 21  *
 22  * A Hotspot is simply an HTML element that is moved / hidden etc.
 23  * to overlay a given position in the panorama.
 24  *
 25  * @param {bigshot​.VRPanorama} panorama the panorama to attach this hotspot to
 26  */
 27 bigshot​.VRHotspot = function (panorama) {
 28     this​.panorama = panorama;
 29     
 30     /**
 31      * The method to use for dealing with hotspots that extend outside the 
 32      * viewport. Note that {@link #CLIP​_ADJUST} et al are functions, not constants.
 33      * To set the value, you must call the function to get a clipping strategy:
 34      *
 35      * @example
 36      * var hotspot = ...;
 37      * // note the function call below ---------------v
 38      * hotspot​.clippingStrategy = hotspot​.CLIP​_ADJUST ();
 39      *
 40      * @see bigshot​.VRHotspot#CLIP​_ADJUST
 41      * @see bigshot​.VRHotspot#CLIP​_CENTER
 42      * @see bigshot​.VRHotspot#CLIP​_FRACTION
 43      * @see bigshot​.VRHotspot#CLIP​_ZOOM
 44      * @see bigshot​.VRHotspot#CLIP​_FADE
 45      * @see bigshot​.VRHotspot#clip
 46      * @type function(clipData)
 47      * @default bigshot​.VRHotspot#CLIP​_ADJUST
 48      */
 49     this​.clippingStrategy = bigshot​.VRHotspot​.CLIP​_ADJUST (panorama);
 50     
 51 }
 52 
 53 /**
 54  * Hides the hotspot if less than <code>frac</code> of its area is visible.
 55  * 
 56  * @param {number} frac the fraction (0​.0 - 1​.0) of the hotspot that must be visible for
 57  * it to be shown.
 58  * @type function(clipData)
 59  * @see bigshot​.VRHotspot#clip
 60  * @see bigshot​.VRHotspot#clippingStrategy
 61  */
 62 bigshot​.VRHotspot​.CLIP​_FRACTION = function (panorama, frac) {
 63     return function (clipData) {
 64         var r = {
 65             x0 : Math​.max (clipData​.x, 0),
 66             y0 : Math​.max (clipData​.y, 0),
 67             x1 : Math​.min (clipData​.x + clipData​.w, panorama​.renderer​.getViewportWidth ()),
 68             y1 : Math​.min (clipData​.y + clipData​.h, panorama​.renderer​.getViewportHeight ())
 69         };
 70         var full = clipData​.w * clipData​.h;
 71         var visibleWidth = (r​.x1 - r​.x0);
 72         var visibleHeight = (r​.y1 - r​.y0);
 73         if (visibleWidth > 0 && visibleHeight > 0) {
 74             var visible = visibleWidth * visibleHeight;
 75             
 76             return (visible / full) >= frac;
 77         } else {
 78             return false;
 79         }
 80     }
 81 };
 82 
 83 /**
 84  * Hides the hotspot if its center is outside the viewport.
 85  * 
 86  * @type function(clipData)
 87  * @see bigshot​.VRHotspot#clip
 88  * @see bigshot​.VRHotspot#clippingStrategy
 89  */
 90 bigshot​.VRHotspot​.CLIP​_CENTER = function (panorama) {
 91     return function (clipData) {
 92         var c = {
 93             x : clipData​.x + clipData​.w / 2,
 94             y : clipData​.y + clipData​.h / 2
 95         };
 96         return c​.x >= 0 && c​.x < panorama​.renderer​.getViewportWidth () && 
 97         c​.y >= 0 && c​.y < panorama​.renderer​.getViewportHeight ();
 98     }
 99 }
100 
101 /**
102  * Resizes the hotspot to fit in the viewport. Hides the hotspot if 
103  * it is completely outside the viewport.
104  * 
105  * @type function(clipData)
106  * @see bigshot​.VRHotspot#clip
107  * @see bigshot​.VRHotspot#clippingStrategy
108  */
109 bigshot​.VRHotspot​.CLIP​_ADJUST = function (panorama) {
110     return function (clipData) {
111         if (clipData​.x < 0) {
112             clipData​.w -= -clipData​.x;
113             clipData​.x = 0;
114         }
115         if (clipData​.y < 0) {
116             clipData​.h -= -clipData​.y;
117             clipData​.y = 0;
118         }
119         if (clipData​.x + clipData​.w > panorama​.renderer​.getViewportWidth ()) {
120             clipData​.w = panorama​.renderer​.getViewportWidth () - clipData​.x - 1;
121         }
122         if (clipData​.y + clipData​.h > panorama​.renderer​.getViewportHeight ()) {
123             clipData​.h = panorama​.renderer​.getViewportHeight () - clipData​.y - 1;
124         }
125         
126         return clipData​.w > 0 && clipData​.h > 0;
127     }
128 }
129 
130 /**
131  * Shrinks the hotspot as it approaches the viewport edges.
132  *
133  * @param s The full size of the hotspot.
134  * @param s​.w The full width of the hotspot, in pixels.
135  * @param s​.h The full height of the hotspot, in pixels.
136  * @see bigshot​.VRHotspot#clip
137  * @see bigshot​.VRHotspot#clippingStrategy
138  */
139 bigshot​.VRHotspot​.CLIP​_ZOOM = function (panorama, s, maxDistanceInViewportHeights) {
140     return function (clipData) {
141         if (clipData​.x >= 0 && clipData​.y >= 0 && (clipData​.x + s​.w) < panorama​.renderer​.getViewportWidth ()
142                 && (clipData​.y + s​.h) < panorama​.renderer​.getViewportHeight ()) {
143                     clipData​.w = s​.w;
144                     clipData​.h = s​.h;
145                     return true;
146                 }
147         
148         var distance = 0;
149         if (clipData​.x < 0) {
150             distance = Math​.max (-clipData​.x, distance);
151         }
152         if (clipData​.y < 0) {
153             distance = Math​.max (-clipData​.y, distance);
154         }
155         if (clipData​.x + s​.w > panorama​.renderer​.getViewportWidth ()) {
156             distance = Math​.max (clipData​.x + s​.w - panorama​.renderer​.getViewportWidth (), distance);
157         }
158         if (clipData​.y + s​.h > panorama​.renderer​.getViewportHeight ()) {
159             distance = Math​.max (clipData​.y + s​.h - panorama​.renderer​.getViewportHeight (), distance);
160         }
161         
162         distance /= panorama​.renderer​.getViewportHeight ();
163         if (distance > maxDistanceInViewportHeights) {
164             return false;
165         }
166         
167         var scale = 1 / (1 + distance);
168         
169         clipData​.w = s​.w * scale;
170         clipData​.h = s​.w * scale;
171         if (clipData​.x < 0) {
172             clipData​.x = 0;
173         }
174         if (clipData​.y < 0) {
175             clipData​.y = 0;
176         }
177         if (clipData​.x + clipData​.w > panorama​.renderer​.getViewportWidth ()) {
178             clipData​.x = panorama​.renderer​.getViewportWidth () - clipData​.w;
179         }
180         if (clipData​.y + clipData​.h > panorama​.renderer​.getViewportHeight ()) {
181             clipData​.y = panorama​.renderer​.getViewportHeight () - clipData​.h;
182         }
183         
184         return true;
185     }
186 }
187 
188 /**
189  * Progressively fades the hotspot as it gets closer to the viewport edges.
190  *
191  * @param {number} borderSizeInPixels the distance from the edge, in pixels,
192  * where the hotspot is completely opaque.
193  * @see bigshot​.VRHotspot#clip
194  * @see bigshot​.VRHotspot#clippingStrategy
195  */
196 bigshot​.VRHotspot​.CLIP​_FADE = function (panorama, borderSizeInPixels) {
197     return function (clipData) {
198         var distance = Math​.min (
199             clipData​.x, 
200             clipData​.y, 
201             panorama​.renderer​.getViewportWidth () - (clipData​.x + clipData​.w), 
202             panorama​.renderer​.getViewportHeight () - (clipData​.y + clipData​.h));
203         
204         if (distance <= 0) {
205             return false;
206         } else if (distance <= borderSizeInPixels) {
207             clipData​.opacity = (distance / borderSizeInPixels);
208             return true;
209         } else {
210             clipData​.opacity = 1​.0;
211             return true;
212         }
213     }
214 }
215 
216 bigshot​.VRHotspot​.prototype = {
217     
218     /**
219      * Layout and resize the hotspot. Called by the panorama.
220      */
221     layout : function () {},
222     
223     /**
224      * Helper function to rotate a point around an axis.
225      *
226      * @param {number} ang the angle
227      * @param {bigshot​.Point3D} vector the vector to rotate around
228      * @param {Vector} point the point
229      * @type Vector
230      * @private
231      */
232     rotate : function (ang, vector, point) {
233         var arad = ang * Math​.PI / 180​.0;
234         var m = Matrix​.Rotation(arad, $V([vector​.x, vector​.y, vector​.z])).ensure4x4 ();
235         return m​.xPoint3Dhom1 (point);
236     },
237     
238     /**
239      * Converts the polar coordinates to world coordinates.
240      * The distance is assumed to be 1​.0.
241      *
242      * @param yaw the yaw, in degrees
243      * @param pitch the pitch, in degrees
244      * @type bigshot​.Point3D
245      */
246     toVector : function (yaw, pitch) {
247         var point = { x : 0, y : 0, z : -1 };
248         point = this​.rotate (-pitch, { x : 1, y : 0, z : 0 }, point);
249         point = this​.rotate (-yaw, { x : 0, y : 1, z : 0 }, point);
250         return point;
251     },
252     
253     /**
254      * Converts the world-coordinate point p to screen coordinates.
255      *
256      * @param {bigshot​.Point3D} p the world-coordinate point
257      * @param {bigshot​.VRPanorama​.CoordinateSystem} [coordinateSystem=bigshot​.VRPanorama​.CoordinateSystem​.IMAGE] the coordinate system
258      * to use.
259      * @type point
260      */
261     toScreen : function (p, coordinateSystem) {
262         var res = this​.panorama​.renderer​.transformToScreen (p, coordinateSystem)
263         return res;
264     },
265     
266     /**
267      * Clips the hotspot against the viewport. Both parameters 
268      * are in​/out. Clipping is done by adjusting the values of the
269      * parameters.
270      * 
271      * @param clipData Information about the hotspot.
272      * @param {number} clipData​.x the x-coordinate of the top-left corner of the hotspot, in pixels.
273      * @param {number} clipData​.y the y-coordinate of the top-left corner of the hotspot, in pixels.
274      * @param {number} clipData​.w the width of the hotspot, in pixels.
275      * @param {number} clipData​.h the height of the hotspot, in pixels.
276      * @param {number} [clipData​.opacity] the opacity of the hotspot, ranging from 0​.0 (transparent) 
277      * to 1​.0 (opaque). If set, the opacity of the hotspot element is adjusted.
278      * @type boolean
279      * @return true if the hotspot is visible, false otherwise
280      */
281     clip : function (clipData) {
282         return this​.clippingStrategy (clipData);
283     }
284 }
