 17 /**
 18  * Creates a new adaptive level-of-detail monitor.
 19  *
 20  * @class An adaptive LOD monitor that adjusts the level of detail of a VR panorama
 21  * to achieve a desired frame rate. To connect it to a VR panorama, use the 
 22  * {@link bigshot.AdaptiveLODMonitor#getListener} method to get a render listener 
 23  * that can be passed to {@link bigshot.VRPanorama#addRenderListener}.
 24  *
 25  * <p>The monitor maintains two render modes - a high quality one with a fixed
 26  * level of detail, and a low(er) quality one with variable level of detail.
 27  * If the panorama is idle for more than a set interval, a high-quality render is
 28  * performed.
 29  * 
 30  * @param {bigshot.AdaptiveLODMonitorParameters} parameters parameters for the LOD monitor.
 31  *
 32  * @see bigshot.AdaptiveLODMonitorParameters for a list of parameters
 33  *
 34  * @example
 35  * var bvr = new bigshot.VRPanorama ( ... );
 36  * var lodMonitor = new bigshot.AdaptiveLODMonitor (
 37  *     new bigshot.AdaptiveLODMonitorParameters ({
 38  *         vrPanorama : bvr,
 39  *         targetFps : 30,
 40  *         tolerance : 0.3,
 41  *         rate : 0.1,
 42  *         minMag : 1.5,
 43  *         maxMag : 16
 44  *     }));
 45  * bvr.addRenderListener (lodMonitor.getListener ());
 46  */
 47 bigshot.AdaptiveLODMonitor = function (parameters) {
 48     this.setParameters (parameters);
 49     
 50     /**
 51      * The current adaptive detail level.
 52      * @type float
 53      * @private
 54      */
 55     this.currentAdaptiveMagnification = parameters.vrPanorama.getMaxTextureMagnification ();
 56     
 57     /**
 58      * The number of frames that have been rendered.
 59      * @type int
 60      * @private
 61      */
 62     this.frames = 0;
 63     
 64     /**
 65      * The total number of times we have sampled the render time.
 66      * @type int
 67      * @private
 68      */
 69     this.samples = 0;
 70     
 71     /**
 72      * The sum of sample times from all samples of render time in milliseconds.
 73      * @type int
 74      * @private
 75      */
 76     this.renderTimeTotal = 0;
 77     
 78     /**
 79      * The sum of sample times from the recent sample pass in milliseconds.
 80      * @type int
 81      * @private
 82      */
 83     this.renderTimeLast = 0;
 84     
 85     /**
 86      * The number of samples currently done in the recent sample pass.
 87      * @type int
 88      * @private
 89      */
 90     this.samplesLast = 0;
 91     
 92     /**
 93      * The start time, in milliseconds, of the last sample.
 94      * @type int
 95      * @private
 96      */
 97     this.startTime = 0;
 98     
 99     /**
100      * The time, in milliseconds, when the panorama was last rendered.
101      * @type int
102      * @private
103      */
104     this.lastRender = 0;
105     
106     this.hqRender = false;
107     this.hqMode = false;
108     this.hqRenderWaiting = false;
109     
110     /**
111      * Flag to enable / disable the monitor.
112      * @type boolean
113      * @private
114      */
115     this.enabled = true;
116     
117     var that = this;
118     this.listenerFunction = function (state, cause, data) {
119         that.listener (state, cause, data);
120     };         
121 };
122 
123 bigshot.AdaptiveLODMonitor.prototype = {
124     averageRenderTime : function () {
125         if (this.samples > 0) {
126             return this.renderTimeTotal / this.samples;
127         } else {
128             return -1;
129         }
130     },
131     
132     /**
133      * @param {bigshot.AdaptiveLODMonitorParameters} parameters
134      */
135     setParameters : function (parameters) {
136         this.parameters = parameters;
137         this.targetTime = 1000 / this.parameters.targetFps;
138         
139         this.lowerTime = this.targetTime / (1.0 + this.parameters.tolerance);
140         this.upperTime = this.targetTime * (1.0 + this.parameters.tolerance);
141     },
142     
143     setEnabled : function (enabled) {
144         this.enabled = enabled;
145     },
146     
147     averageRenderTimeLast : function () {
148         if (this.samples > 0) {
149             return this.renderTimeLast / this.samplesLast;
150         } else {
151             return -1;
152         }
153     },
154     
155     getListener : function () {
156         return this.listenerFunction;
157     },
158     
159     increaseDetail : function () {
160         this.currentAdaptiveMagnification = Math.max (this.parameters.minMag, this.currentAdaptiveMagnification / (1.0 + this.parameters.rate));
161     },
162     
163     decreaseDetail : function () {
164         this.currentAdaptiveMagnification = Math.min (this.parameters.maxMag, this.currentAdaptiveMagnification * (1.0 + this.parameters.rate));
165     },
166     
167     sample : function () {
168         var deltat = new Date ().getTime () - this.startTime;
169         this.samples++;
170         this.renderTimeTotal += deltat;
171         
172         this.samplesLast++;
173         this.renderTimeLast += deltat;
174         
175         if (this.samplesLast > 4) {
176             var averageLast = this.renderTimeLast / this.samplesLast;                        
177             
178             if (averageLast < this.lowerTime) {
179                 this.increaseDetail ();
180             } else if (averageLast > this.upperTime) {
181                 this.decreaseDetail ();
182             }
183             
184             this.samplesLast = 0;
185             this.renderTimeLast = 0;
186         }
187     },
188     
189     hqRenderTick : function () {
190         if (this.lastRender < new Date ().getTime () - this.parameters.hqRenderDelay) {
191             this.hqRender = true;
192             this.hqMode = true;
193             if (this.enabled) {
194                 this.parameters.vrPanorama.setMaxTextureMagnification (this.parameters.hqRenderMag);
195                 this.parameters.vrPanorama.render ();
196             }
197             
198             this.hqRender = false;
199             this.hqRenderWaiting = false;
200         } else {
201             var that = this;
202             setTimeout (function () {
203                     that.hqRenderTick ();
204                 }, this.parameters.hqRenderInterval);
205         }
206     },
207     
208     listener : function (state, cause, data) {
209         if (!this.enabled) {
210             return;
211         }
212         
213         if (this.hqRender) {
214             return;
215         }
216         
217         if (this.hqMode && cause == bigshot.VRPanorama.ONRENDER_TEXTURE_UPDATE) {
218             this.parameters.vrPanorama.setMaxTextureMagnification (this.parameters.minMag);
219             return;
220         } else {
221             this.hqMode = false;
222         }
223         
224         this.parameters.vrPanorama.setMaxTextureMagnification (this.currentAdaptiveMagnification);
225         
226         this.frames++;
227         if ((this.frames < 20 || this.frames % 5 == 0) && state == bigshot.VRPanorama.ONRENDER_BEGIN) {
228             this.startTime = new Date ().getTime ();
229             this.lastRender = this.startTime;
230             var that = this;
231             setTimeout (function () {
232                     that.sample ();
233                 }, 1);
234             if (!this.hqRenderWaiting) {
235                 this.hqRenderWaiting = true;
236                 setTimeout (function () {
237                         that.hqRenderTick ();
238                     }, this.parameters.hqRenderInterval);
239             }
240         }
241     }
242 };
