 17 /**
 18  * Creates a new instance of a folder-based filesystem adapter.
 19  *
 20  * @augments bigshot.FileSystem
 21  * @class Folder-based filesystem.
 22  * @param {bigshot.ImageParameters|bigshot.VRPanoramaParameters} parameters the associated image parameters
 23  * @constructor
 24  */
 25 bigshot.FolderFileSystem = function (parameters) {
 26     this.prefix = null;
 27     this.suffix = "";
 28     this.parameters = parameters;
 29 }
 30 
 31 
 32 bigshot.FolderFileSystem.prototype = {    
 33     getDescriptor : function () {
 34         this.browser = new bigshot.Browser ();
 35         var req = this.browser.createXMLHttpRequest ();
 36         
 37         req.open("GET", this.getFilename ("descriptor"), false);   
 38         req.send(null); 
 39         var descriptor = {};
 40         if(req.status == 200) {
 41             var substrings = req.responseText.split (":");
 42             for (var i = 0; i < substrings.length; i += 2) {
 43                 if (substrings[i] == "suffix") {
 44                     descriptor[substrings[i]] = substrings[i + 1];
 45                 } else {
 46                     descriptor[substrings[i]] = parseInt (substrings[i + 1]);
 47                 }
 48             }
 49             this.suffix = descriptor.suffix;
 50             return descriptor;
 51         } else {
 52             throw new Error ("Unable to find descriptor.");
 53         }
 54     },
 55     
 56     getPosterFilename : function () {
 57         return this.getFilename ("poster" + this.suffix);
 58     },
 59     
 60     setPrefix : function (prefix) {
 61         this.prefix = prefix;
 62     },
 63     
 64     getPrefix : function () {
 65         if (this.prefix) {
 66             return this.prefix + "/";
 67         } else {
 68             return "";
 69         }
 70     },
 71     
 72     getFilename : function (name) {
 73         return this.parameters.basePath + "/" + this.getPrefix () + name;
 74     },
 75     
 76     getImageFilename : function (tileX, tileY, zoomLevel) {
 77         var key = (-zoomLevel) + "/" + tileX + "_" + tileY + this.suffix;
 78         return this.getFilename (key);
 79     }
 80 };
 81 
 82 bigshot.Object.validate ("bigshot.FolderFileSystem", bigshot.FileSystem);
