 18  * Abstract filesystem definition.
 19  *
 20  * @class Abstract filesystem definition.

 22 bigshot​.FileSystem = function () {
 23 }

 25 bigshot​.FileSystem​.prototype = {
 26     /**
 27      * Returns the URL filename for the given filesystem entry.
 28      *
 29      * @param {String} name the entry name
 30      */
 31     getFilename : function (name) {},
 32     
 33     /**
 34      * Returns the entry filename for the given tile.
 35      * 
 36      * @param {int} tileX the column of the tile
 37      * @param {int} tileY the row of the tile
 38      * @param {int} zoomLevel the zoom level
 39      */
 40     getImageFilename : function (tileX, tileY, zoomLevel) {},
 41     
 42     /**
 43      * Sets an optional prefix that is prepended, along with a forward
 44      * slash ("/"), to all names.
 45      *
 46      * @param {String} prefix the prefix
 47      */
 48     setPrefix : function (prefix) {},
 49     
 50     /**
 51      * Returns an image descriptor object from the descriptor file.
 52      *
 53      * @param {Function} [callback] an optional callback that receives the descriptor

 56      */
 57     getDescriptor : function (callback) {},

 59     /**
 60      * Returns the poster URL filename. For Bigshot images this is
 61      * typically the URL corresponding to the entry "poster​.jpg", 
 62      * but for other filesystems it can be different.

 64     getPosterFilename : function () {}
 65 };

 72 /**
 73  * Sets up a filesystem instance in the given parameters object, if none exist.
 74  * If the {@link bigshot​.ImageParameters#fileSystem} member isn't set, the 
 75  * {@link bigshot​.ImageParameters#fileSystemType} member is used to create a new 
 76  * {@link bigshot​.FileSystem} instance and set it.
 77  *
 78  * @param {bigshot​.ImageParameters or bigshot​.VRPanoramaParameters or bigshot​.ImageCarouselPanoramaParameters} parameters the parameters object to populate
 79  */
 80 bigshot​.setupFileSystem = function (parameters) {
 81     if (!parameters​.fileSystem) {
 82         if (parameters​.fileSystemType == "archive") {
 83             parameters​.fileSystem = new bigshot​.ArchiveFileSystem (parameters);
 84         } else if (parameters​.fileSystemType == "dzi") {
 85             parameters​.fileSystem = new bigshot​.DeepZoomImageFileSystem (parameters);
 86         } else if (parameters​.fileSystemType == "simple") {
 87             parameters​.fileSystem = new bigshot​.SimpleFileSystem (parameters);
 88         } else {
 89             parameters​.fileSystem = new bigshot​.FolderFileSystem (parameters);
 90         }
 91     }
 92 }
