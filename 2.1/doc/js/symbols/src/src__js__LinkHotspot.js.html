 17 #include LabeledHotspot​.js
 18 #include Object​.js
 19     
 20 /**
 21  * Creates a new link-hotspot instance.
 22  *
 23  * @class A labeled hotspot that takes the user to another
 24  * location when it is clicked on. See {@link bigshot​.HotspotLayer} for 
 25  * examples.
 26  *
 27  * @see bigshot​.HotspotLayer
 28  * @param {number} x x-coordinate of the top-left corner, given in full image pixels
 29  * @param {number} y y-coordinate of the top-left corner, given in full image pixels
 30  * @param {number} w width of the hotspot, given in full image pixels
 31  * @param {number} h height of the hotspot, given in full image pixels
 32  * @param {String} labelText text of the label
 33  * @param {String} url url to go to on click
 34  * @augments bigshot​.LabeledHotspot
 35  * @constructor
 36  */
 37 bigshot​.LinkHotspot = function (x, y, w, h, labelText, url) {
 38     bigshot​.LabeledHotspot​.call (this, x, y, w, h, labelText);
 39     this​.browser​.registerListener (this​.getElement (), "click", function () {
 40             document​.location​.href = url;
 41         });
 42 };
 43 
 44 bigshot​.Object​.extend (bigshot​.LinkHotspot, bigshot​.LabeledHotspot);
