 17 /**
 18  * Creates a textured quad scene.
 19  *
 20  * @param {bigshot.WebGL} webGl the webGl instance to use for rendering.
 21  *
 22  * @class A "scene" consisting of a number of quads, all with
 23  * a unique texture. Used by the {@link bigshot.VRPanorama} to render the VR cube.
 24  *
 25  * @see bigshot.WebGLTexturedQuad
 26  */
 27 bigshot.WebGLTexturedQuadScene = function (webGl, buffers) {
 28     this.quads = new Array ();
 29     this.webGl = webGl;
 30     this.buffers = buffers;
 31 }
 32 
 33 bigshot.WebGLTexturedQuadScene.prototype = {
 34     /** 
 35      * Adds a new quad to the scene.
 36      */
 37     addQuad : function (quad) {
 38         this.quads.push (quad);
 39     },
 40     
 41     /** 
 42      * Renders all quads.
 43      */
 44     render : function () {
 45         var b = this.buffers.get ();
 46         var vertexPositionBuffer = b.vertexPositionBuffer;
 47         var textureCoordBuffer = b.textureCoordBuffer;
 48         var vertexIndexBuffer = b.vertexIndexBuffer;
 49         
 50         this.webGl.setMatrixUniforms();
 51         
 52         for (var i = 0; i < this.quads.length; ++i) {
 53             this.quads[i].render (this.webGl, vertexPositionBuffer, textureCoordBuffer, vertexIndexBuffer);
 54         }
 55     }
 56 };
