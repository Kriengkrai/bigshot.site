 17 /**
 18  * Creates a new filesystem instance.
 19  *
 20  * @class Abstract base class for filesystems.
 21  */
 22 bigshot​.AbstractFileSystem = function () {
 23     
 24 }
 25 
 26 bigshot​.AbstractFileSystem​.prototype = {
 27     mediaTypesForExtension : {
 28         "jpg" : "image",
 29         "jpeg" : "image",
 30         "png" : "image",
 31         "gif" : "image",
 32         
 33         "webm" : "video",
 34         "mp4" : "video",
 35         "m4v" : "video",
 36         "flv" : "video",
 37         "mpg" : "video",        
 38         "avi" : "video"
 39     },
 40     
 41     getMediaTypeForExtension : function (extension) {
 42         return this​.mediaTypesForExtension[extension​.toLowerCase ()];
 43     },
 44     
 45     chooseMediaFormat : function (formatString, altFormats) {
 46         return {
 47             suffix : formatString,
 48             type : this​.getMediaTypeForExtension (formatString)
 49         };
 50     }    
 51 };
