 20 /**
 21  * Creates an image event.
 22  *
 23  * @class Base class for events dispatched by bigshot​.VRPanorama.
 24  * @param {Object} data a data object whose fields will be used to set the 
 25  * corresponding fields of the event object.
 26  * @extends bigshot​.Event
 27  * @see bigshot​.VRPanorama
 28  */
 29 bigshot​.VREvent = function (data) {
 30     bigshot​.Event​.call (this, data);
 31 }
