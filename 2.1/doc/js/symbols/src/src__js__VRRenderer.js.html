 17 /**
 18  * @class Abstract base for 3d rendering system.
 19  */
 20 bigshot​.VRRenderer = function () {
 21 }
 22 
 23 bigshot​.VRRenderer​.prototype = {
 24     /**
 25      * Creates a new {@link bigshot​.VRTileCache}, appropriate for the rendering system.
 26      *
 27      * @param {function()} onloaded function that is called whenever a texture tile has been loaded
 28      * @param {function()} onCacheInit function that is called when the texture cache is fully initialized
 29      * @param {bigshot​.VRPanoramaParameters} parameters the parameters for the panorama
 30      */
 31     createTileCache : function (onloaded, onCacheInit, parameters) {},
 32     
 33     /**
 34      * Creates a bigshot​.TexturedQuadScene.
 35      */
 36     createTexturedQuadScene : function () {},
 37     
 38     /**
 39      * Creates a bigshot​.TexturedQuad.
 40      *
 41      * @param {bigshot​.Point3D} p the top-left corner of the quad
 42      * @param {bigshot​.Point3D} u a vector going along the top edge of the quad
 43      * @param {bigshot​.Point3D} v a vector going down the left edge of the quad
 44      * @param {Object} texture a texture to use for the quad. The texture type may vary among different
 45      * VRRenderer implementations. The VRTileCache that is created using the createTileCache method will
 46      * supply the correct type.
 47      */
 48     createTexturedQuad : function (p, u, v, texture) {},
 49     
 50     /**
 51      * Returns the viewport width, in pixels.
 52      *
 53      * @type int
 54      */
 55     getViewportWidth : function () {},
 56     
 57     /**
 58      * Returns the viewport height, in pixels.
 59      *
 60      * @type int
 61      */
 62     getViewportHeight : function () {},
 63     
 64     /**
 65      * Transforms a vector to world coordinates.
 66      *
 67      * @param {bigshot​.Point3D} v the view-space point to transform
 68      */
 69     transformToWorld : function (v) {},
 70     
 71     /**
 72      * Transforms a world vector to screen coordinates.
 73      *
 74      * @param {bigshot​.Point3D} worldVector the world-space point to transform
 75      */
 76     transformWorldToScreen : function (worldVector) {},
 77     
 78     /**
 79      * Transforms a 3D vector to screen coordinates.
 80      *
 81      * @param {bigshot​.Point3D} vector the vector to transform. 
 82      * If it is already in homogenous coordinates (4-element array) 
 83      * the transformation is faster. Otherwise it will be converted.
 84      */
 85     transformToScreen : function (vector) {},
 86     
 87     /**
 88      * Disposes the renderer and associated resources.
 89      */
 90     dispose : function () {},
 91     
 92     /**
 93      * Called to begin a render.
 94      *
 95      * @param {bigshot​.Rotation} rotation the rotation of the viewer
 96      * @param {number} fov the vertical field of view, in degrees
 97      * @param {bigshot​.Point3D} translation the position of the viewer in world space
 98      * @param {bigshot​.Rotation} rotationOffsets the rotation to apply to the VR cube 
 99      * before the viewer rotation is applied
100      */
101     beginRender : function (rotation, fov, translation, rotationOffsets) {},
102     
103     /**
104      * Called to end a render.
105      */
106     endRender : function () {},
107     
108     /**
109      * Called by client code to notify the renderer that the viewport has been resized.
110      */
111     onresize : function () {},
112     
113     /**
114      * Resizes the viewport.
115      *
116      * @param {int} w the new width of the viewport, in pixels
117      * @param {int} h the new height of the viewport, in pixels
118      */
119     resize : function (w, h) {},
120     
121     /**
122      * Gets the container element for the renderer. This is used
123      * when calling the requestAnimationFrame API.
124      */
125     getElement : function () {}
126 }
