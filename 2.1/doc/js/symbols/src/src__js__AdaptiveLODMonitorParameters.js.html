 17 /**
 18  * Creates a new parameter block.
 19  *
 20  * @class Parameters for the adaptive LOD monitor.
 21  */
 22 bigshot​.AdaptiveLODMonitorParameters = function (values) {
 23     
 24     /**
 25      * The VR panorama to adjust.
 26      *
 27      * @type bigshot​.VRPanorama
 28      */
 29     this​.vrPanorama = null;
 30     
 31     /**
 32      * The target framerate in frames per second. 
 33      * The monitor will try to achieve an average frame render time
 34      * of <i>1 / targetFps</i> seconds.
 35      *
 36      * @default 30
 37      * @type float
 38      */
 39     this​.targetFps = 30;
 40     
 41     /**
 42      * The tolerance for the rendering time. The monitor will adjust the
 43      * level of detail if the average frame render time rises above
 44      * <i>target frame render time * (1​.0 + tolerance)</i> or falls below
 45      * <i>target frame render time / (1​.0 + tolerance)</i>.
 46      *
 47      * @default 0​.3
 48      * @type float
 49      */
 50     this​.tolerance = 0​.3;
 51     
 52     /**
 53      * The rate at which the level of detail is adjusted.
 54      * For detail increase, the detail is multiplied with (1​.0 + rate),
 55      * for decrease divided.
 56      *
 57      * @default 0​.1
 58      * @type float
 59      */
 60     this​.rate = 0​.1;
 61     
 62     /**
 63      * Minimum texture magnification.
 64      *
 65      * @default 1​.5
 66      * @type float
 67      */
 68     this​.minMag = 1​.5;
 69     
 70     /**
 71      * Maximum texture magnification.
 72      *
 73      * @default 16
 74      * @type float     
 75      */
 76     this​.maxMag = 16;
 77     
 78     /**
 79      * Texture magnification for HQ render passes.
 80      *
 81      * @default 1​.5
 82      * @type float     
 83      */
 84     this​.hqRenderMag = 1​.5;
 85     
 86     /**
 87      * Delay in milliseconds before executing 
 88      * a HQ render pass.
 89      *
 90      * @default 2000
 91      * @type int
 92      */
 93     this​.hqRenderDelay = 2000;
 94     
 95     /**
 96      * Interval in milliseconds for the 
 97      * HQ render pass timer.
 98      *
 99      * @default 1000
100      * @type int
101      */
102     this​.hqRenderInterval = 1000;
103     
104     if (values) {
105         for (var k in values) {
106             this[k] = values[k];
107         }
108     }
109     
110     this​.merge = function (values, overwrite) {
111         for (var k in values) {
112             if (overwrite || !this[k]) {
113                 this[k] = values[k];
114             }
115         }
116     }
117     return this;        
118 };
