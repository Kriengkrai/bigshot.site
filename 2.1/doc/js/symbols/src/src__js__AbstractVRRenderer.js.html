 17 #include VRPanorama​.js
 18     
 19 /**
 20  * @class Abstract VR renderer base class.
 21  */
 22 bigshot​.AbstractVRRenderer = function () {
 23 }
 24 
 25 bigshot​.AbstractVRRenderer​.prototype = {
 26     /**
 27      * Transforms a vector to world coordinates.
 28      *
 29      * @param {bigshot​.Point3D} vector the vector to transform
 30      */
 31     transformToWorld : function transformToWorld (vector) {
 32         var world = this​.mvMatrix​.matrix ().xPoint3Dhom1 (vector);
 33         
 34         return world;
 35     },
 36     
 37     /**
 38      * Transforms a world vector to screen coordinates.
 39      *
 40      * @param {bigshot​.Point3D} world the world-vector to transform
 41      */
 42     transformWorldToScreen : function transformWorldToScreen (world) {
 43         if (world​.z > 0) {
 44             return null;
 45         }
 46         
 47         var screen = this​.pMatrix​.matrix ().xPoint3Dhom (world);
 48         if (Math​.abs (screen​.w) < Sylvester​.precision) {
 49             return null;
 50         }
 51         
 52         var sx = screen​.x;
 53         var sy = screen​.y;
 54         var sz = screen​.z;
 55         var vw = this​.getViewportWidth ();
 56         var vh = this​.getViewportHeight ();
 57         
 58         var r = {
 59             x: (vw / 2) * sx / sz + vw / 2, 
 60             y: - (vh / 2) * sy / sz + vh / 2
 61         };
 62         return r;
 63     },
 64     
 65     /**
 66      * Transforms a vector to screen coordinates.
 67      *
 68      * @param {bigshot​.Point3D} vector the vector to transform
 69      * @param {bigshot​.VRPanorama​.CoordinateSystem} [coordinateSystem=bigshot​.VRPanorama​.CoordinateSystem​.IMAGE] the coordinate system
 70      * to use.
 71      * @return the transformed vector, or null if the vector is nearer than the near-z plane.
 72      */
 73     transformToScreen : function transformToScreen (vector, coordinateSystem) {
 74         var sel = null;
 75         if (coordinateSystem == bigshot​.VRPanorama​.CoordinateSystem​.VIEW) {
 76             sel = this​.vpMatrix​.xPoint3Dhom (vector);
 77         } else {
 78             sel = this​.mvpMatrix​.xPoint3Dhom (vector);
 79         }
 80         
 81         if (sel​.z < 0) {
 82             return null;
 83         }
 84         
 85         var sz = sel​.w;
 86         
 87         if (Math​.abs (sel​.w) < Sylvester​.precision) {
 88             return null;
 89         }
 90         
 91         var sx = sel​.x;
 92         var sy = sel​.y;
 93         var vw = this​.getViewportWidth ();
 94         var vh = this​.getViewportHeight ();
 95         
 96         var r = {
 97             x: (vw / 2) * sx / sz + vw / 2, 
 98             y: - (vh / 2) * sy / sz + vh / 2
 99         };
100 
101         return r;
102     }
103 }
