 17 /**
 18  * Creates an event.
 19  *
 20  * @class Base class for events. The interface is supposed to be as similar to
 21  * standard DOM events as possible.
 22  * @param {Object} data a data object whose fields will be used to set the 
 23  * corresponding fields of the event object.
 24  */
 25 bigshot.Event = function (data) {
 26 
 27     /**
 28      * Indicates whether the event bubbles.
 29      * @default false
 30      * @type boolean
 31      */
 32     this.bubbles = false;
 33     
 34     /**
 35      * Indicates whether the event is cancelable.
 36      * @default false
 37      * @type boolean
 38      */
 39     this.cancelable = false;
 40     
 41     /**
 42      * The current target of the event
 43      * @default null
 44      */
 45     this.currentTarget = null;
 46     
 47     /**
 48      * Set if the preventDefault method has been called.
 49      * @default false
 50      * @type boolean
 51      */
 52     this.defaultPrevented = false;
 53 
 54     /**
 55      * The target to which the event is dispatched.
 56      * @default null
 57      */
 58     this.target = null;
 59     
 60     /**
 61      * The time the event was created, in milliseconds since the epoch.
 62      * @default the current time, as given by <code>new Date ().getTime ()</code>
 63      * @type number
 64      */
 65     this.timeStamp = new Date ().getTime ();
 66     
 67     /**
 68      * The event type.
 69      * @default null
 70      * @type String
 71      */
 72     this.type = null;
 73     
 74     /**
 75      * Flag indicating origin of event.
 76      * @default false
 77      * @type boolean
 78      */
 79     this.isTrusted = false;
 80     
 81     for (var k in data) {
 82         this[k] = data[k];
 83     }
 84 }
 85 
 86 bigshot.Event.prototype = {
 87     /**
 88      * Prevents default handling of the event.
 89      */
 90     preventDefault : function () {
 91         this.defaultPrevented = true;
 92     }
 93 };
