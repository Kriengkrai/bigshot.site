 17 /**
 18  * Creates a new browser helper object.
 19  *
 20  * @class Encapsulates common browser functions for cross-browser portability
 21  * and convenience.
 22  */
 23 bigshot.Browser = function () {
 24     this.requestAnimationFrameFunction = 
 25         window.requestAnimationFrame || 
 26         window.mozRequestAnimationFrame ||  
 27         window.webkitRequestAnimationFrame || 
 28         window.msRequestAnimationFrame ||
 29         function (callback, element) { return setTimeout (callback, 0); };
 30 }
 31 
 32 bigshot.Browser.prototype = {
 33     /**
 34     * Removes all children from an element.
 35     * 
 36     * @public
 37     * @param {HTMLElement} element the element whose children are to be removed.
 38     */
 39     removeAllChildren : function (element) {
 40         element.innerHTML = "";
 41         /*

 43             for (var i = element.children.length - 1; i >= 0; --i) {
 44                 element.removeChild (element.children[i]);
 45             }
 46         }
 47         */
 48     },
 49     
 50     /**
 51     * Thunk to implement a faked "mouseenter" event.
 52     * @private
 53     */
 54     mouseEnter : function (_fn) {
 55         var isAChildOf = this.isAChildOf;
 56         return function(_evt)
 57         {
 58             var relTarget = _evt.relatedTarget;
 59             if (this === relTarget || isAChildOf (this, relTarget))
 60             { return; }
 61             
 62             _fn.call (this, _evt);
 63         }
 64     },
 65     
 66     isAChildOf : function (_parent, _child) {
 67         if (_parent === _child) { return false; }
 68         while (_child && _child !== _parent)
 69         { _child = _child.parentNode; }
 70         
 71         return _child === _parent;
 72     },
 73     
 74     /**
 75     * Unregisters a listener from an element.
 76     *
 77     * @param {HTMLElement} elem the element
 78     * @param {String} eventName the event name ("click", "mouseover", etc.)
 79     * @param {function(e)} fn the callback function to detach
 80     * @param {boolean} useCapture specifies if we should unregister a listener from the capture chain.
 81     */
 82     unregisterListener : function (elem, eventName, fn, useCapture) {
 83         if (typeof (elem.removeEventListener) != 'undefined') {
 84             elem.removeEventListener (eventName, fn, useCapture);
 85         } else if (typeof (elem.detachEvent) != 'undefined') {
 86             elem.detachEvent('on' + eventName, fn);
 87         }
 88     },
 89     
 90     /**
 91     * Registers a listener to an element.
 92     *
 93     * @param {HTMLElement} elem the element
 94     * @param {String} eventName the event name ("click", "mouseover", etc.)
 95     * @param {function(e)} fn the callback function to attach
 96     * @param {boolean} useCapture specifies if we want to initiate capture.
 97     * See <a href="https://developer.mozilla.org/en/DOM/element.addEventListener">element.addEventListener</a>
 98     * on MDN for an explanation.
 99     */
100     registerListener : function (_elem, _evtName, _fn, _useCapture) {
101         if (typeof _elem.addEventListener != 'undefined')
102         {
103             if (_evtName === 'mouseenter')
104             { _elem.addEventListener('mouseover', this.mouseEnter(_fn), _useCapture); }
105             else if (_evtName === 'mouseleave')
106             { _elem.addEventListener('mouseout', this.mouseEnter(_fn), _useCapture); }
107             else
108             { _elem.addEventListener(_evtName, _fn, _useCapture); }
109         }
110         else if (typeof _elem.attachEvent != 'undefined')
111         {
112             _elem.attachEvent('on' + _evtName, _fn);
113         }
114         else
115         {
116             _elem['on' + _evtName] = _fn;
117         }
118     },
119     
120     /**
121     * Stops an event from bubbling.
122     *
123     * @param {Event} eventObject the event object
124     */
125     stopEventBubbling : function (eventObject) {
126         if (eventObject) {
127             if (eventObject.stopPropagation) {
128                 eventObject.stopPropagation ();
129             } else { 
130                 eventObject.cancelBubble = true; 
131             }
132         }
133     },
134     
135     /**
136     * Creates a callback function that simply stops the event from bubbling.
137     *
138     * @example
139     * var browser = new bigshot.Browser ();
140     * browser.registerListener (element, 
141     *     "mousedown", 
142     *     browser.stopEventBubblingHandler (), 
143     *     false);
144     * @type function(event)
145     * @return a new function that can be used to stop an event from bubbling
146     */
147     stopEventBubblingHandler : function () {
148         var that = this;
149         return function (event) {
150             that.stopEventBubbling (event);
151             return false;
152         };
153     },
154     
155     /**
156         * Stops bubbling for all mouse events on the element.
157         *
158         * @param {HTMLElement} element the element
159         */
160     stopMouseEventBubbling : function (element) {
161         this.registerListener (element, "mousedown", this.stopEventBubblingHandler (), false);
162         this.registerListener (element, "mouseup", this.stopEventBubblingHandler (), false);
163         this.registerListener (element, "mousemove", this.stopEventBubblingHandler (), false);
164     },
165     
166     /**
167      * Returns the size in pixels of the element
168      *
169      * @param {HTMLElement} obj the element
170      * @return a size object with two integer members, w and h, for width and height respectively.
171      */
172     getElementSize : function (obj) {
173         var size = {};
174         if (obj.clientWidth) {
175             size.w = obj.clientWidth;
176         }
177         if (obj.clientHeight) {
178             size.h = obj.clientHeight;
179         }
180         return size;
181     },
182     
183     /**
184      * Returns true if the browser is scaling the window, such as on Mobile Safari.
185      * The method used here is far from perfect, but it catches the most important use case:
186      * If we are running on an iDevice and the page is zoomed out.
187      */
188     browserIsViewporting : function () {
189         if (window.innerWidth <= screen.width) {
190             return false;
191         } else {
192             return true;
193         }
194     },
195     
196     /**
197      * Returns the device pixel scale, which is equal to the number of device 
198      * pixels each css pixel corresponds to. Used to render the proper level of detail
199      * on mobile devices, especially when zoomed out and more detailed textures are
200      * simply wasted.
201      */
202     getDevicePixelScale : function () {
203         if (this.browserIsViewporting ()) {
204             return screen.width / window.innerWidth;
205         } else {
206             return 1.0;
207         }
208     },
209     
210     requestAnimationFrame : function (callback, element) {
211         var raff = this.requestAnimationFrameFunction;
212         raff (callback, element);
213     },
214     
215     /**
216         * Returns the position in pixels of the element relative
217         * to the top left corner of the document.
218         *
219         * @param {HTMLElement} obj the element
220         * @return a position object with two integer members, x and y.
221         */
222     getElementPosition : function (obj) {
223         var position = new Object();
224         position.x = 0;
225         position.y = 0;
226         
227         var o = obj;
228         while (o) {
229             position.x += o.offsetLeft;
230             position.y += o.offsetTop;
231             if (o.clientLeft) {
232                 position.x += o.clientLeft;
233             }
234             if (o.clientTop) {
235                 position.y += o.clientTop;
236             }
237             
238             if (o.x) {
239                 position.x += o.x;
240             }
241             if (o.y) {
242                 position.y += o.y;
243             }
244             o = o.offsetParent;
245         }
246         return position;
247     },
248     
249     /**
250      * Creates an XMLHttpRequest object.
251      *
252      * @type XMLHttpRequest
253      * @return a XMLHttpRequest object.
254      */
255     createXMLHttpRequest : function  () {
256         try { 
257             return new ActiveXObject("Msxml2.XMLHTTP"); 
258         } catch (e) {
259         }
260         
261         try { 
262             return new ActiveXObject("Microsoft.XMLHTTP"); 
263         } catch (e) {
264         }
265         
266         try { 
267             return new XMLHttpRequest(); 
268         } catch(e) {
269         }
270         
271         alert("XMLHttpRequest not supported");
272         
273         return null;
274     }
275 };
