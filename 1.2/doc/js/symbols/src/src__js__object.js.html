 17 /**
 18  * @class Object-oriented support functions, used to make JavaScript
 19  * a bit more palatable to a Java-head.
 20  */
 21 bigshot.object = {
 22     /**
 23      * Extends a base class with a derived class.
 24      *
 25      * @param {Function} derived the derived-class
 26      * @param {Function} base the base-class
 27      */
 28     extend : function (derived, base) {
 29         for (var k in base.prototype) {
 30             if (derived.prototype[k]) {
 31                 derived.prototype[k]._super = base.prototype[k];
 32             } else {
 33                 derived.prototype[k] = base.prototype[k];
 34             }
 35         }
 36     },
 37     
 38     /**
 39      * Resolves a name relative to <code>self</code>.
 40      *
 41      * @param {String} name the name to resolve
 42      * @type {Object}
 43      */
 44     resolve : function (name) {
 45         var c = name.split (".");
 46         var clazz = self;
 47         for (var i = 0; i < c.length; ++i) {
 48             clazz = clazz[c[i]];
 49         }
 50         return clazz;
 51     },
 52     
 53     validate : function (clazzName, iface) {
 54         #ifdef DEBUG
 55             console.log ("Validating " + clazzName);
 56         var clazz = this.resolve (clazzName);
 57         for (var k in iface.prototype) {
 58             if (!clazz.prototype[k]) {
 59                 throw new Error (clazzName + " doesn't implement " + k);
 60             }
 61         }
 62         #endif
 63     },
 64     
 65     /**
 66      * Utility function to show an object's fields in a message box.
 67      *
 68      * @param {Object} o the object
 69      */
 70     alertr : function (o) {
 71         var sb = "";
 72         for (var k in o) {
 73             sb += k + ":" + o[k] + "\n";
 74         }
 75         alert (sb);
 76     },
 77     
 78     /**
 79      * Utility function to show an object's fields in the console log.
 80      *
 81      * @param {Object} o the object
 82      */
 83     logr : function (o) {
 84         var sb = "";
 85         for (var k in o) {
 86             sb += k + ":" + o[k] + "\n";
 87         }
 88         if (console) {
 89             console.log (sb);
 90         }
 91     }
 92 };
