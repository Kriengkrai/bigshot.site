 18 /**
 19  * Creates a new tiled image viewer. (Note: See {@link bigshot.ImageBase#dispose} for important information.)
 20  *
 21  * @example
 22  * var bsi = new bigshot.Image (
 23  *     new bigshot.ImageParameters ({
 24  *         basePath : "/bigshot.php?file=myshot.bigshot",
 25  *         fileSystemType : "archive",
 26  *         container : document.getElementById ("bigshot_div")
 27  *     }));
 28  *
 29  * @param {bigshot.ImageParameters} parameters the image parameters. Required fields are: <code>basePath</code> and <code>container</code>.
 30  * If you intend to use the archive filesystem, you need to set the <code>fileSystemType</code> to <code>"archive"</code>
 31  * as well.
 32  * @see bigshot.ImageBase#dispose
 33  * @class A tiled, zoomable image viewer.
 34  *
 35  * <h3 id="creating-a-wrapping-image">Creating a Wrapping Image</h3>
 36  *
 37  * <p>If you have set the wrapX or wrapY parameters in the {@link bigshot.ImageParameters}, the 
 38  * image must be an integer multiple of the tile size at the desired minimum zoom level, otherwise
 39  * there will be a gap at the wrap point:
 40  *
 41  * <p>The way to figure out the proper input size is this:
 42  *
 43  * <ol>
 44  * <li><p>Decide on a tile size and call this <i>tileSize</i>.</p></li>
 45  * <li><p>Decide on a minimum integer zoom level, and call this <i>minZoom</i>.</p></li>
 46  * <li><p>Compute <i>tileSize * 2<sup>-minZoom</sup></i>, call this <i>S</i>.</p></li>
 47  * <li><p>The source image size along the wrapped axis must be evenly divisible by <i>S</i>.</p></li>
 48  * </ol>
 49  *
 50  * <p>An example:</p>
 51  *
 52  * <ol>
 53  * <li><p>I have an image that is 23148x3242 pixels.</p></li>
 54  * <li><p>I chose 256x256 pixel tiles: <i>tileSize = 256</i>.</p></li>
 55  * <li><p>When displaying the image, I want the user to be able to zoom out so that the 
 56  * whole image is less than or equal to 600 pixels tall. Since the image is 3242 pixels 
 57  * tall originally, I will need a <i>minZoom</i> of -3. A <i>minZoom</i> of -2 would only let me
 58  * zoom out to 1/4 (2<sup>-2</sup>), or an image that is 810 pixels tall. A <i>minZoom</i> of -3, however lets me
 59  * zoom out to 1/8 (2<sup>-3</sup>), or an image that is 405 pixels tall. Thus: <i>minZoom = -3</i></p></li>
 60  * <li><p>Computing <i>S</i> gives: <i>S = 256 * 2<sup>3</sup> = 256 * 8 = 2048</i></p></li>
 61  * <li><p>I want it to wrap along the X axis. Therefore I may have to adjust the width, 
 62  * currently 23148 pixels.</p></li>
 63  * <li><p>Rounding 23148 down to the nearest multiple of 2048 gives 22528. (23148 divided by 2048 is 11.3, and 11 times 2048 is 22528.)</p></li>
 64  * <li><p>I will shrink my source image to be 22528 pixels wide before building the image pyramid,
 65  * and I will set the <code>minZoom</code> parameter to -3 in the {@link bigshot.ImageParameters} when creating
 66  * the image. (I will also set <code>wrapX</code> to <code>true</code>.)</p></li>
 67  * </ol>
 68  * 
 69  * @augments bigshot.ImageBase
 70  */     
 71 bigshot.Image = function (parameters) {
 72     bigshot.setupFileSystem (parameters);
 73     parameters.merge (parameters.fileSystem.getDescriptor (), false);
 74     
 75     bigshot.ImageBase.call (this, parameters);
 76 }    
 77 
 78 bigshot.Image.prototype = {
 79     setupLayers : function () {
 80         var that = this;
 81         this.thisTileCache = new bigshot.ImageTileCache (function () {
 82                 that.layout ();     
 83             }, null, this.parameters);
 84         
 85         this.addLayer (
 86             new bigshot.TileLayer (this, this.parameters, 0, 0, this.thisTileCache)
 87         );
 88     }
 89 };
 90 
 91 bigshot.object.extend (bigshot.Image, bigshot.ImageBase);
