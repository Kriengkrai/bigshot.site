 17 /**
 18  * Creates a new image parameter object and populates it with default values for
 19  * all values not explicitly given.
 20  *
 21  * @class ImageParameters parameter object.
 22  * You need not set any fields that can be read from the image descriptor that 
 23  * MakeImagePyramid creates. See the {@link bigshot.Image} documentation for 
 24  * required parameters.
 25  *
 26  * <p>Usage:
 27  *
 28  * @example
 29  * var bsi = new bigshot.Image (
 30  *     new bigshot.ImageParameters ({
 31  *         basePath : "/bigshot.php?file=myshot.bigshot",
 32  *         fileSystemType : "archive",
 33  *         container : document.getElementById ("bigshot_div")
 34  *         }));
 35  * 
 36  * @param values named parameter map, see the fields below for parameter names and types.
 37  * @see bigshot.Image
 38  */
 39 bigshot.ImageParameters = function (values) {
 40     /**
 41      * Size of low resolution preview image along the longest image
 42      * dimension. The preview is assumed to have the same aspect
 43      * ratio as the full image (specified by width and height).
 44      *
 45      * @default <i>Optional</i> set by MakeImagePyramid and loaded from descriptor
 46      * @type int
 47      * @public
 48      */
 49     this.posterSize = 0;
 50     
 51     /**
 52      * Url for the image tile to show while the tile is loading and no 
 53      * low-resolution preview is available.
 54      *
 55      * @default <code>null</code>, which results in an all-black image
 56      * @type String
 57      * @public
 58      */
 59     this.emptyImage = null;
 60     
 61     /**
 62      * Suffix to append to the tile filenames. Typically <code>".jpg"</code> or 
 63      * <code>".png"</code>.
 64      *
 65      * @default <i>Optional</i> set by MakeImagePyramid and loaded from descriptor
 66      * @type String
 67      */
 68     this.suffix = null;
 69     
 70     /**
 71      * The width of the full image; in pixels.
 72      *
 73      * @default <i>Optional</i> set by MakeImagePyramid and loaded from descriptor
 74      * @type int
 75      */
 76     this.width = 0;
 77     
 78     /**
 79      * The height of the full image; in pixels.
 80      *
 81      * @default <i>Optional</i> set by MakeImagePyramid and loaded from descriptor
 82      * @type int
 83      */
 84     this.height = 0;
 85     
 86     /**
 87      * For {@link bigshot.Image}, the {@code div} to use as a container for the image.
 88      *
 89      * @type HTMLDivElement
 90      */
 91     this.container = null;
 92     
 93     /**
 94      * The minimum zoom value. Zoom values are specified as a magnification; where
 95      * 2<sup>n</sup> is the magnification and n is the zoom value. So a zoom value of
 96      * 2 means a 4x magnification of the full image. -3 means showing an image that
 97      * is a eighth (1/8 or 1/2<sup>3</sup>) of the full size.
 98      *
 99      * @type number
100      * @default <i>Optional</i> set by MakeImagePyramid and loaded from descriptor
101      */
102     this.minZoom = 0.0;
103     
104     /**
105      * The maximum zoom value. Zoom values are specified as a magnification; where
106      * 2<sup>n</sup> is the magnification and n is the zoom value. So a zoom value of
107      * 2 means a 4x magnification of the full image. -3 means showing an image that
108      * is a eighth (1/8 or 1/2<sup>3</sup>) of the full size.
109      *
110      * @type number
111      * @default 0.0
112      */
113     this.maxZoom = 0.0;
114     
115     /**
116      * Size of one tile in pixels.
117      *
118      * @type int
119      * @default <i>Optional</i> set by MakeImagePyramid and loaded from descriptor
120      */
121     this.tileSize = 0;
122     
123     /**
124      * Tile overlap. Not implemented.
125      *
126      * @type int
127      * @default <i>Optional</i> set by MakeImagePyramid and loaded from descriptor
128      */
129     this.overlap = 0;
130     
131     /**
132      * Flag indicating that the image should wrap horizontally. The image wraps on tile
133      * boundaries; so in order to get a seamless wrap at zoom level -n; the image width must
134      * be evenly divisible by <code>tileSize * 2^n</code>. Set the minZoom value appropriately.
135      * 
136      * @type boolean
137      * @default false
138      */
139     this.wrapX = false;
140     
141     /**
142      * Flag indicating that the image should wrap vertically. The image wraps on tile
143      * boundaries; so in order to get a seamless wrap at zoom level -n; the image height must
144      * be evenly divisible by <code>tileSize * 2^n</code>. Set the minZoom value appropriately.
145      *
146      * @type boolean
147      * @default false
148      */
149     this.wrapY = false;
150     
151     /**
152      * Base path for the image. This is filesystem dependent; but for the two most common cases
153      * the following should be set
154      *
155      * <ul>
156      * <li><b>archive</b>= The basePath is <code>"<path>/bigshot.php?file=<path-to-bigshot-archive-relative-to-bigshot.php>"</code>;
157      *     for example; <code>"/bigshot.php?file=images/bigshot-sample.bigshot"</code>.
158      * <li><b>folder</b>= The basePath is <code>"<path-to-image-folder>"</code>;
159      *     for example; <code>"/images/bigshot-sample"</code>.
160      * </ul>
161      *
162      * @type String
163      */
164     this.basePath = null;
165     
166     /**
167      * The file system type. Used to create a filesystem instance unless
168      * the fileSystem field is set. Possible values are <code>"archive"</code>, 
169      * <code>"folder"</code> or <code>"dzi"</code>.
170      *
171      * @type String
172      * @default "folder"
173      */
174     this.fileSystemType = "folder";
175     
176     /**
177      * A reference to a filesystem implementation. If set; it overrides the
178      * fileSystemType field.
179      *
180      * @default set depending on value of bigshot.ImageParameters.fileSystemType
181      * @type bigshot.FileSystem
182      */
183     this.fileSystem = null;
184     
185     /**
186      * Object used to load data files.
187      *
188      * @default bigshot.DefaultDataLoader
189      * @type bigshot.DataLoader
190      */
191     this.dataLoader = new bigshot.DefaultDataLoader ();
192     
193     /**
194      * Enable the touch-friendly ui. The touch-friendly UI splits the viewport into
195      * three click-sensitive regions:
196      * <p style="text-align:center"><img src="../images/touch-ui.png"/></p>
197      * 
198      * <p>Clicking (or tapping with a finger) on the outer region causes the viewport to zoom out.
199      * Clicking anywhere within the middle, "pan", region centers the image on the spot clicked.
200      * Finally, clicking in the center hotspot will center the image on the spot clicked and zoom
201      * in half a zoom level.
202      *
203      * <p>As before, you can drag to pan anywhere.
204      *
205      * <p>If you have navigation tools for mouse users that hover over the image container, it 
206      * is recommended that any click events on them are kept from bubbling, otherwise the click 
207      * will propagate to the touch ui. One way is to use the 
208      * {@link bigshot.Browser#stopMouseEventBubbling} method:
209      *
210      * @example
211      * var browser = new bigshot.Browser ();
212      * browser.stopMouseEventBubbling (document.getElementById ("myBigshotControlDiv"));
213      *
214      * @see bigshot.Image#showTouchUI
215      *
216      * @type boolean
217      * @default true
218      * @deprecated Bigshot supports all common touch-gestures.
219      */
220     this.touchUI = false;
221     
222     /**
223      * Lets you "fling" the image.
224      * 
225      * @type boolean
226      * @default true
227      */
228     this.fling = true;
229     
230     /**
231      * The maximum amount that a tile will be stretched until we try to show
232      * the next more detailed level.
233      *
234      * @type float
235      * @default 1.0
236      */
237     this.maxTextureMagnification = 1.0;
238     
239     if (values) {
240         for (var k in values) {
241             this[k] = values[k];
242         }
243     }
244     
245     this.merge = function (values, overwrite) {
246         for (var k in values) {
247             if (overwrite || !this[k]) {
248                 this[k] = values[k];
249             }
250         }
251     }
252     return this;        
253 };
