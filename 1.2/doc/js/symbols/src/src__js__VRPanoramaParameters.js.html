 17 /**
 18  * Creates a new VR panorama parameter object and populates it with default values for
 19  * all values not explicitly given.
 20  *
 21  * @class VRPanoramaParameters parameter object.
 22  * You need not set any fields that can be read from the image descriptor that 
 23  * MakeImagePyramid creates. See the {@link bigshot.VRPanorama}
 24  * documentation for required parameters.
 25  *
 26  * <p>Usage:
 27  *
 28  * @example
 29  * var bvr = new bigshot.VRPanorama (
 30  *     new bigshot.VRPanoramaParameters ({
 31  *         basePath : "/bigshot.php?file=myvr.bigshot",
 32  *         fileSystemType : "archive",
 33  *         container : document.getElementById ("bigshot_canvas")
 34  *         }));
 35  * @param values named parameter map, see the fields below for parameter names and types.
 36  * @see bigshot.VRPanorama
 37  */
 38 bigshot.VRPanoramaParameters = function (values) {
 39     /**
 40      * Size of low resolution preview image along the longest image
 41      * dimension. The preview is assumed to have the same aspect
 42      * ratio as the full image (specified by width and height).
 43      *
 44      * @default <i>Optional</i> set by MakeImagePyramid and loaded from descriptor
 45      * @type int
 46      * @public
 47      */
 48     this.posterSize = 0;
 49     
 50     /**
 51      * Url for the image tile to show while the tile is loading and no 
 52      * low-resolution preview is available.
 53      *
 54      * @default <code>null</code>, which results in an all-black image
 55      * @type String
 56      * @public
 57      */
 58     this.emptyImage = null;
 59     
 60     /**
 61      * Suffix to append to the tile filenames. Typically <code>".jpg"</code> or 
 62      * <code>".png"</code>.
 63      *
 64      * @default <i>Optional</i> set by MakeImagePyramid and loaded from descriptor
 65      * @type String
 66      */
 67     this.suffix = null;
 68     
 69     /**
 70      * The width of the full image; in pixels.
 71      *
 72      * @default <i>Optional</i> set by MakeImagePyramid and loaded from descriptor
 73      * @type int
 74      */
 75     this.width = 0;
 76     
 77     /**
 78      * The height of the full image; in pixels.
 79      *
 80      * @default <i>Optional</i> set by MakeImagePyramid and loaded from descriptor
 81      * @type int
 82      */
 83     this.height = 0;
 84     
 85     /**
 86      * For {@link bigshot.VRPanorama}, the {@code div} to render into.
 87      *
 88      * @type HTMLDivElement
 89      */
 90     this.container = null;
 91     
 92     /**
 93      * The maximum number of times to split a cube face into four quads.
 94      *
 95      * @type int
 96      * @default <i>Optional</i> set by MakeImagePyramid and loaded from descriptor
 97      */
 98     this.maxTesselation = -1;
 99     
100     /**
101      * Size of one tile in pixels.
102      *
103      * @type int
104      * @default <i>Optional</i> set by MakeImagePyramid and loaded from descriptor
105      */
106     this.tileSize = 0;
107     
108     /**
109      * Tile overlap. Not implemented.
110      *
111      * @type int
112      * @default <i>Optional</i> set by MakeImagePyramid and loaded from descriptor
113      */
114     this.overlap = 0;
115     
116     /**
117      * Base path for the image. This is filesystem dependent; but for the two most common cases
118      * the following should be set
119      *
120      * <ul>
121      * <li><b>archive</b>= The basePath is <code>"<path>/bigshot.php?file=<path-to-bigshot-archive-relative-to-bigshot.php>"</code>;
122      *     for example; <code>"/bigshot.php?file=images/bigshot-sample.bigshot"</code>.
123      * <li><b>folder</b>= The basePath is <code>"<path-to-image-folder>"</code>;
124      *     for example; <code>"/images/bigshot-sample"</code>.
125      * </ul>
126      *
127      * @type String
128      */
129     this.basePath = null;
130     
131     /**
132      * The file system type. Used to create a filesystem instance unless
133      * the fileSystem field is set. Possible values are <code>"archive"</code>, 
134      * <code>"folder"</code> or <code>"dzi"</code>.
135      *
136      * @type String
137      * @default "folder"
138      */
139     this.fileSystemType = "folder";
140     
141     /**
142      * A reference to a filesystem implementation. If set; it overrides the
143      * fileSystemType field.
144      *
145      * @default set depending on value of bigshot.VRPanoramaParameters#fileSystemType
146      * @type bigshot.FileSystem
147      */
148     this.fileSystem = null;
149     
150     /**
151      * Object used to load data files.
152      *
153      * @default bigshot.DefaultDataLoader
154      * @type bigshot.DataLoader
155      */
156     this.dataLoader = new bigshot.DefaultDataLoader ();
157     
158     /**
159      * The maximum magnification for the texture tiles making up the VR cube.
160      * Used for level-of-detail tesselation.
161      * A value of 1.0 means that textures will never be stretched (one texture pixel will
162      * always be at most one screen pixel), unless there is no more detailed texture available. 
163      * A value of 2.0 means that textures may be stretched at most 2x (one texture pixel 
164      * will always be at most 2x2 screen pixels)
165      * The bigger the value, the less texture data is required, but quality suffers.
166      *
167      * @type number
168      * @default 1.0
169      */
170     this.maxTextureMagnification = 1.0;
171     
172     /**
173      * The WebGL texture filter to use for magnifying textures. 
174      * Possible values are all values valid for <code>TEXTURE_MAG_FILTER</code>.
175      * <code>null</code> means <code>NEAREST</code>. 
176      *
177      * @default null / NEAREST.
178      */
179     this.textureMagFilter = null;
180     
181     /**
182      * The WebGL texture filter to use for supersampling (minifying) textures. 
183      * Possible values are all values valid for <code>TEXTURE_MIN_FILTER</code>.
184      * <code>null</code> means <code>NEAREST</code>. 
185      *
186      * @default null / NEAREST.
187      */
188     this.textureMinFilter = null;
189     
190     /**
191      * Minimum vertical field of view in degrees.
192      *
193      * @default 2.0
194      * @type number
195      */
196     this.minFov = 2.0;
197     
198     /**
199      * Maximum vertical field of view in degrees.
200      *
201      * @default 90.0
202      * @type number
203      */
204     this.maxFov = 90;
205     
206     /**
207      * Minimum pitch in degrees.
208      *
209      * @default -90
210      * @type number
211      */
212     this.minPitch = -90;
213     
214     /**
215      * Maximum pitch in degrees.
216      *
217      * @default 90.0
218      * @type number
219      */
220     this.maxPitch = 90;
221     
222     /**
223      * Minimum yaw in degrees. The number is interpreted modulo 360.
224      * The default value, -360, is just to make sure that we won't accidentally
225      * trip it. If the number is set to something in the interval 0-360,
226      * the autoRotate function will pan back and forth.
227      *
228      * @default -360
229      * @type number
230      */
231     this.minYaw = -360;
232     
233     /**
234      * Maximum yaw in degrees. The number is interpreted modulo 360.
235      * The default value, 720, is just to make sure that we won't accidentally
236      * trip it. If the number is set to something in the interval 0-360,
237      * the autoRotate function will pan back and forth.
238      *
239      * @default 720.0
240      * @type number
241      */
242     this.maxYaw = 720;
243     
244     /**
245      * Transform offset for yaw.
246      * @default 0.0
247      * @type number
248      */
249     this.yawOffset = 0.0;
250     
251     /**
252      * Transform offset for pitch.
253      * @default 0.0
254      * @type number
255      */
256     this.pitchOffset = 0.0;
257     
258     /**
259      * Transform offset for roll.
260      * @default 0.0
261      * @type number
262      */
263     this.rollOffset = 0.0;
264     
265     /**
266      * Function to call when all six cube faces have loaded the base texture level
267      * and can be rendered.
268      *
269      * @type function()
270      * @default null
271      */
272     this.onload = null;
273     
274     /**
275      * The rendering back end to use.
276      * Values are "css" and "webgl".
277      * 
278      * @type String
279      * @default null
280      */
281     this.renderer = null;
282     
283     /**
284      * Controls whether the panorama can be "flung" by quickly dragging and releasing.
285      * 
286      * @type boolean 
287      * @default true
288      */
289     this.fling = true;
290     
291     /**
292      * Controls the decay of the "flinging" animation. The fling animation decays
293      * as 2^(flingScale * t) where t is the time in milliseconds since the animation started.
294      * For the animation to decay to half-speed in X seconds,
295      * flingScale should then be set to 1 / (X*1000).
296      *
297      * @type float
298      * @default 0.004
299      */
300     this.flingScale = 0.004;
301     
302     if (values) {
303         for (var k in values) {
304             this[k] = values[k];
305         }
306     }
307     
308     this.merge = function (values, overwrite) {
309         for (var k in values) {
310             if (overwrite || !this[k]) {
311                 this[k] = values[k];
312             }
313         }
314     }
315     return this;        
316 };
