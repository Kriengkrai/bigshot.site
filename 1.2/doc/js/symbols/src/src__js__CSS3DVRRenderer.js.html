 17 /**
 18  * @class CSS 3D Transform-based renderer.
 19  *
 20  * @augments bigshot.VRRenderer
 21  */
 22 bigshot.CSS3DVRRenderer = function (_container) {
 23     this.container = _container;
 24     this.canvasOrigin = document.createElement ("div");
 25     
 26     this.canvasOrigin.style.WebkitTransformOrigin = "0px 0px 0px";
 27     this.canvasOrigin.style.WebkitTransformStyle = "preserve-3d";
 28     this.canvasOrigin.style.WebkitPerspective= "600px";
 29     
 30     this.canvasOrigin.style.position = "relative";
 31     this.canvasOrigin.style.left = "50%";
 32     this.canvasOrigin.style.top = "50%";
 33     
 34     this.container.appendChild (this.canvasOrigin);
 35     
 36     this.viewport = document.createElement ("div");
 37     this.viewport.style.WebkitTransformOrigin = "0px 0px 0px";
 38     this.viewport.style.WebkitTransformStyle = "preserve-3d";
 39     this.canvasOrigin.appendChild (this.viewport);
 40     
 41     this.world = document.createElement ("div");
 42     this.world.style.WebkitTransformOrigin = "0px 0px 0px";
 43     this.world.style.WebkitTransformStyle = "preserve-3d";
 44     this.viewport.appendChild (this.world);
 45     
 46     this.browser.removeAllChildren (this.world);
 47     
 48     this.view = null;
 49     
 50     this.mvMatrix = new bigshot.TransformStack ();
 51     
 52     this.yaw = 0;
 53     this.pitch = 0;
 54     this.fov = 0;
 55     this.pMatrix = new bigshot.TransformStack ();
 56     
 57     this.onresize = function () {
 58     };
 59     
 60     this.viewportSize = null;
 61 };
 62 
 63 bigshot.CSS3DVRRenderer.prototype = {
 64     browser : new bigshot.Browser (),
 65     
 66     createTileCache : function (onloaded, onCacheInit, parameters) {
 67         return new bigshot.ImageVRTileCache (onloaded, onCacheInit, parameters);
 68     },
 69     
 70     createTexturedQuadScene : function () {
 71         return new bigshot.CSS3DTexturedQuadScene (this.world, 128, this.view);
 72     },
 73     
 74     createTexturedQuad : function (p, u, v, texture) {
 75         return new bigshot.CSS3DTexturedQuad (p, u, v, texture);
 76     },
 77     
 78     getElement : function () {
 79         return this.container;
 80     },
 81     
 82     supportsUpdate : function () {
 83         return false;
 84     },
 85     
 86     getViewportWidth : function () {
 87         if (this.viewportSize) {
 88             return this.viewportSize.w;
 89         }
 90         return this.browser.getElementSize (this.container).w;
 91     },
 92     
 93     getViewportHeight : function () {
 94         if (this.viewportSize) {
 95             return this.viewportSize.h;
 96         }
 97         return this.browser.getElementSize (this.container).h;
 98     },
 99     
100     resize : function (w, h) {
101         if (this.container.style.width != "") {
102             this.container.style.width = w + "px";
103         }
104         if (this.container.style.height != "") {
105             this.container.style.height = h + "px";
106         }
107     },
108     
109     beginRender : function (y, p, fov, tx, ty, tz, oy, op, or) {
110         this.viewportSize = this.browser.getElementSize (this.container);
111         
112         this.yaw = y;
113         this.pitch = p;
114         this.fov = fov;
115         
116         var halfFovInRad = 0.5 * fov * Math.PI / 180;
117         var halfHeight = this.getViewportHeight () / 2;
118         var perspectiveDistance = halfHeight / Math.tan (halfFovInRad);
119         
120         this.mvMatrix.reset ();
121         
122         this.mvMatrix.translate ([tx, ty, tz]);
123         this.view = {
124             x : tx,
125             y : ty,
126             z : tz
127         };
128         
129         this.mvMatrix.rotate (or, [0, 0, 1]);
130         this.mvMatrix.rotate (op, [1, 0, 0]);
131         this.mvMatrix.rotate (oy, [0, 1, 0]);
132         
133         this.mvMatrix.rotate (this.yaw, [0, 1, 0]);
134         this.mvMatrix.rotate (this.pitch, [1, 0, 0]);
135         
136         
137         this.pMatrix.reset ();
138         this.pMatrix.perspective (this.fov, this.getViewportWidth () / this.getViewportHeight (), 0.1, 100.0);
139         
140         this.mvpMatrix = this.pMatrix.matrix ().multiply (this.mvMatrix.matrix ());
141         
142         this.canvasOrigin.style.WebkitPerspective= perspectiveDistance + "px";
143         
144         for (var i = this.world.children.length - 1; i >= 0; --i) {
145             this.world.children[i].inWorld = 1;
146         }
147         
148         this.world.style.WebkitTransform = 
149             "rotate3d(1,0,0," + (-p) + "deg) " +
150             "rotate3d(0,1,0," + y + "deg) " +
151             "rotate3d(0,1,0," + (oy) + "deg) " +
152             "rotate3d(1,0,0," + (-op) + "deg) " +
153             "rotate3d(0,0,1," + (-or) + "deg) ";
154         this.world.style.WebkitTransformStyle = "preserve-3d";
155         this.world.style.WebKitBackfaceVisibility = "hidden";
156         
157         this.viewport.style.WebkitTransform = 
158             "translateZ(" + perspectiveDistance + "px)";
159     },
160     
161     endRender : function () {
162         for (var i = this.world.children.length - 1; i >= 0; --i) {
163             var child = this.world.children[i];
164             if (!child.inWorld || child.inWorld != 2) {
165                 delete child.inWorld;
166                 this.world.removeChild (child);
167             }
168         }
169         
170         this.viewportSize = null;
171     }    
172 };
173 
174 bigshot.object.extend (bigshot.CSS3DVRRenderer, bigshot.AbstractVRRenderer);
175 bigshot.object.validate ("bigshot.CSS3DVRRenderer", bigshot.VRRenderer);
