 17 /**
 18  * Creates an image event.
 19  *
 20  * @class Base class for events dispatched by bigshot.VRPanorama.
 21  * @param {Object} data a data object whose fields will be used to set the 
 22  * corresponding fields of the event object.
 23  * @extends bigshot.Event
 24  * @see bigshot.VRPanorama
 25  */
 26 bigshot.VREvent = function (data) {
 27     bigshot.Event.call (this, data);
 28 }
