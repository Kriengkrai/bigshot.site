 17 /**
 18  * Creates a new hotspot instance.
 19  *
 20  * @class Base class for hotspots in a {@link bigshot.HotspotLayer}. See {@link bigshot.HotspotLayer} for 
 21  * examples.
 22  *
 23  * @param {number} x x-coordinate of the top-left corner, given in full image pixels
 24  * @param {number} y y-coordinate of the top-left corner, given in full image pixels
 25  * @param {number} w width of the hotspot, given in full image pixels
 26  * @param {number} h height of the hotspot, given in full image pixels
 27  * @see bigshot.HotspotLayer
 28  * @see bigshot.LabeledHotspot
 29  * @see bigshot.LinkHotspot
 30  * @constructor
 31  */
 32 bigshot.Hotspot = function (x, y, w, h) {
 33     var element = document.createElement ("div");
 34     element.style.position = "absolute";
 35     element.style.overflow = "visible";
 36     
 37     this.element = element;
 38     this.x = x;
 39     this.y = y;
 40     this.w = w;
 41     this.h = h;
 42 }
 43 
 44 bigshot.Hotspot.prototype = {
 45     
 46     browser : new bigshot.Browser (),
 47     
 48     /**
 49      * Lays out the hotspot in the viewport.
 50      *
 51      * @name bigshot.Hotspot#layout
 52      * @param x0 x-coordinate of top-left corner of the full image in css pixels
 53      * @param y0 y-coordinate of top-left corner of the full image in css pixels
 54      * @param zoomFactor the zoom factor.
 55      * @function
 56      */
 57     layout : function (x0, y0, zoomFactor) {
 58         var sx = this.x * zoomFactor + x0;
 59         var sy = this.y * zoomFactor + y0;
 60         var sw = this.w * zoomFactor;
 61         var sh = this.h * zoomFactor;
 62         this.element.style.top = sy + "px";
 63         this.element.style.left = sx + "px";
 64         this.element.style.width = sw + "px";
 65         this.element.style.height = sh + "px";
 66     },
 67     
 68     /**
 69      * Returns the HTMLDivElement used to show the hotspot.
 70      * Clients can access this element in order to style it.
 71      *
 72      * @type HTMLDivElement
 73      */
 74     getElement : function () {
 75         return this.element;
 76     }
 77 };
