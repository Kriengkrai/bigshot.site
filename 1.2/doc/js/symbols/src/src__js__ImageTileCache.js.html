 17 /**
 18  * Creates a new cache instance.
 19  *
 20  * @class Tile cache for the {@link bigshot.TileLayer}.
 21  * @constructor
 22  */
 23 bigshot.ImageTileCache = function (onLoaded, onCacheInit, parameters) {
 24     var that = this;
 25     
 26     this.parameters = parameters;
 27     
 28     /**
 29      * Reduced-resolution preview of the full image.
 30      * Loaded from the "poster" image created by 
 31      * MakeImagePyramid
 32      *
 33      * @private
 34      * @type HTMLImageElement
 35      */
 36     this.fullImage = null;
 37     parameters.dataLoader.loadImage (parameters.fileSystem.getPosterFilename (), function (tile) {
 38             that.fullImage = tile;
 39             if (onCacheInit) {
 40                 onCacheInit ();
 41             }
 42         });
 43     
 44     /**
 45      * Maximum number of tiles in the cache.
 46      * @private
 47      * @type int
 48      */
 49     this.maxCacheSize = 512;
 50     this.maxTileX = 0;
 51     this.maxTileY = 0;
 52     this.cachedImages = {};
 53     this.requestedImages = {};
 54     this.usedImages = {};
 55     this.lastOnLoadFiredAt = 0;
 56     this.imageRequests = 0;
 57     this.lruMap = new bigshot.LRUMap ();
 58     this.onLoaded = onLoaded;
 59     this.browser = new bigshot.Browser ();
 60     this.partialImageSize = parameters.tileSize / 4;
 61     this.POSTER_ZOOM_LEVEL = Math.log (parameters.posterSize / Math.max (parameters.width, parameters.height)) / Math.log (2);
 62 }
 63 
 64 bigshot.ImageTileCache.prototype = {
 65     resetUsed : function () {
 66         this.usedImages = {};
 67     },
 68     
 69     setMaxTiles : function (mtx, mty) {
 70         this.maxTileX = mtx;
 71         this.maxTileY = mty;
 72     },
 73     
 74     getPartialImage : function (tileX, tileY, zoomLevel) {
 75         var img = this.getPartialImageFromDownsampled (tileX, tileY, zoomLevel, 0, 0, this.parameters.tileSize, this.parameters.tileSize);
 76         if (img == null) {
 77             img = this.getPartialImageFromPoster (tileX, tileY, zoomLevel);
 78         }
 79         return img;
 80     },
 81     
 82     getPartialImageFromPoster : function (tileX, tileY, zoomLevel) {
 83         if (this.fullImage && this.fullImage.complete) {
 84             var posterScale = this.fullImage.width / this.parameters.width;
 85             var tileSizeAtZoom = posterScale * this.parameters.tileSize / Math.pow (2, zoomLevel);
 86             
 87             x0 = Math.floor (tileSizeAtZoom * tileX);
 88             y0 = Math.floor (tileSizeAtZoom * tileY);
 89             w = Math.floor (tileSizeAtZoom);
 90             h = Math.floor (tileSizeAtZoom);
 91             
 92             return this.createPartialImage (this.fullImage, this.fullImage.width, x0, y0, w, h);
 93         } else {
 94             return null;
 95         }
 96     },
 97     
 98     createPartialImage : function (sourceImage, expectedSourceImageSize, x0, y0, w, h) {
 99         var canvas = document.createElement ("canvas");
100         if (!canvas["width"]) {
101             return null;
102         }
103         canvas.width = this.partialImageSize;
104         canvas.height = this.partialImageSize;
105         var ctx = canvas.getContext('2d'); 
106         
107         var scale = sourceImage.width / expectedSourceImageSize;
108         
109         var sx = Math.floor (x0 * scale);
110         var sy = Math.floor (y0 * scale);
111         w *= scale;
112         if (sx + w >= sourceImage.width) {
113             w = sourceImage.width - sx - 1;
114         }
115         
116         h *= scale;
117         if (sy + h >= sourceImage.height) {
118             h = sourceImage.height - sy - 1;
119         }
120         
121         ctx.drawImage (sourceImage, sx, sy, w, h, -0.1, -0.1, this.partialImageSize + 0.2, this.partialImageSize + 0.2);
122         
123         return canvas;
124     },
125     
126     getPartialImageFromDownsampled : function (tileX, tileY, zoomLevel, x0, y0, w, h) {
127         var key = this.getImageKey (tileX, tileY, zoomLevel);
128         var sourceImage = this.cachedImages[key];
129         
130         if (sourceImage == null) {
131             this.requestImage (tileX, tileY, zoomLevel);
132             
133             // Give up if the poster image has higher resolution.
134             if (zoomLevel < this.POSTER_ZOOM_LEVEL) {
135                 return null;
136             }
137         }
138         
139         if (sourceImage) {
140             return this.createPartialImage (sourceImage, this.parameters.tileSize, x0, y0, w, h);
141         } else {
142             w /= 2;
143             h /= 2;
144             x0 /= 2;
145             y0 /= 2;
146             if ((tileX % 2) == 1) {
147                 x0 += this.parameters.tileSize / 2;
148             }
149             if ((tileY % 2) == 1) {
150                 y0 += this.parameters.tileSize / 2;
151             }
152             tileX = Math.floor (tileX / 2);
153             tileY = Math.floor (tileY / 2);
154             --zoomLevel;
155             return this.getPartialImageFromDownsampled (tileX, tileY, zoomLevel, x0, y0, w, h);
156         }        
157     },
158     
159     getEmptyImage : function () {
160         var tile = document.createElement ("img");
161         if (this.parameters.emptyImage) {
162             tile.src = this.parameters.emptyImage;
163         } else {
164             tile.src = "data:image/gif,GIF89a%01%00%01%00%80%00%00%00%00%00%FF%FF%FF!%F9%04%00%00%00%00%00%2C%00%00%00%00%01%00%01%00%00%02%02D%01%00%3B";
165         }
166         return tile;
167     },
168     
169     getImage : function (tileX, tileY, zoomLevel) {
170         if (tileX < 0 || tileY < 0 || tileX >= this.maxTileX || tileY >= this.maxTileY) {
171             return this.getEmptyImage ();
172         }
173         
174         var key = this.getImageKey (tileX, tileY, zoomLevel);
175         this.lruMap.access (key);
176         
177         if (this.cachedImages[key]) {
178             if (this.usedImages[key]) {
179                 var tile = this.parameters.dataLoader.loadImage (this.getImageFilename (tileX, tileY, zoomLevel));
180                 return tile;
181             } else {
182                 this.usedImages[key] = true;
183                 return this.cachedImages[key];
184             }
185         } else {
186             var img = this.getPartialImage (tileX, tileY, zoomLevel);
187             if (img != null) {
188                 this.cachedImages[key] = img;
189             } else {
190                 img = this.getEmptyImage ();
191             }
192             
193             return img;
194         }
195     },
196     
197     requestImage : function (tileX, tileY, zoomLevel) {
198         var key = this.getImageKey (tileX, tileY, zoomLevel);
199         if (!this.requestedImages[key]) {
200             this.imageRequests++;
201             var that = this;
202             this.parameters.dataLoader.loadImage (this.getImageFilename (tileX, tileY, zoomLevel), function (tile) {
203                     that.cachedImages[key] = tile;
204                     delete that.requestedImages[key];
205                     that.imageRequests--;
206                     var now = new Date();
207                     if (that.imageRequests == 0 || now.getTime () > (that.lastOnLoadFiredAt + 50)) {
208                         that.purgeCache ();
209                         that.lastOnLoadFiredAt = now.getTime ();
210                         that.onLoaded ();
211                     }
212                 });
213             this.requestedImages[key] = true;
214         }            
215     },
216     
217     /**
218      * Removes the least-recently used objects from the cache,
219      * if the size of the cache exceeds the maximum cache size.
220      * A maximum of four objects will be removed per call.
221      *
222      * @private
223      */
224     purgeCache : function () {
225         for (var i = 0; i < 4; ++i) {
226             if (this.lruMap.getSize () > this.maxCacheSize) {
227                 var leastUsed = this.lruMap.leastUsed ();
228                 this.lruMap.remove (leastUsed);
229                 delete this.cachedImages[leastUsed];                    
230             }
231         }
232     },
233     
234     getImageKey : function (tileX, tileY, zoomLevel) {
235         return "I" + tileX + "_" + tileY + "_" + zoomLevel;
236     },
237     
238     getImageFilename : function (tileX, tileY, zoomLevel) {
239         var f = this.parameters.fileSystem.getImageFilename (tileX, tileY, zoomLevel);
240         return f;
241     }
242 };
