 17 /**
 18  * Creates a new HTML element layer. The layer must be added to the image using
 19  * {@link bigshot.ImageBase#addLayer}.
 20  *
 21  * @class A layer consisting of a single HTML element that is moved and scaled to cover
 22  * the layer.
 23  * @example
 24  * var image = new bigshot.Image (...);
 25  * image.addLayer (
 26  *     new bigshot.HTMLElementLayer (this, this.imgElement, this.parameters.width, this.parameters.height)
 27  * );
 28  * @param {bigshot.ImageBase} image the image this hotspot layer will be part of
 29  * @param {HTMLElement} element the element to present in this layer
 30  * @param {int} width the width, in image pixels (display size at zoom level 0), of the HTML element
 31  * @param {int} height the height, in image pixels (display size at zoom level 0), of the HTML element
 32  * @augments bigshot.Layer
 33  */
 34 bigshot.HTMLDivElementLayer = function (image, element, width, height, wrapX, wrapY) {
 35     this.wrapX = wrapX;
 36     this.wrapY = wrapY;
 37     this.hotspots = new Array ();
 38     this.browser = new bigshot.Browser ();
 39     this.image = image;
 40     this.container = image.createLayerContainer ();
 41     this.parentContainer = image.getContainer ();
 42     this.element = element;
 43     this.parentContainer.appendChild (element);
 44     this.w = width;
 45     this.h = height;
 46     this.resize (0, 0);
 47 }
 48 
 49 bigshot.HTMLDivElementLayer.prototype = {
 50     
 51     getContainer : function () {
 52         return this.container;
 53     },
 54     
 55     resize : function (w, h) {
 56         this.container.style.width = this.parentContainer.clientWidth + "px";
 57         this.container.style.height = this.parentContainer.clientHeight + "px";
 58     },
 59     
 60     layout : function (zoom, x0, y0, tx0, ty0, size, stride, opacity) {
 61         var zoomFactor = Math.pow (2, this.image.getZoom ());
 62         x0 -= stride * tx0;
 63         y0 -= stride * ty0;
 64         
 65         var imW = (this.w * zoomFactor);
 66         var imH = (this.h * zoomFactor);
 67         
 68         this.element.style.backgroundSize = imW + "px " + imH + "px";
 69             
 70         var bposX = "0px";
 71         var bposY = "0px";
 72         
 73         if (this.wrapY) {
 74             this.element.style.top = "0px";
 75             this.element.style.height = (this.parentContainer.clientHeight) + "px";
 76             bposY = y0 + "px";
 77         } else {
 78             this.element.style.top = y0 + "px";
 79             this.element.style.height = imH + "px";
 80         }
 81         
 82         if (this.wrapX) {
 83             this.element.style.left = "0px";
 84             this.element.style.width = (this.parentContainer.clientWidth) + "px";            
 85             bposX = x0 + "px";
 86         } else {
 87             this.element.style.left = x0 + "px";
 88             this.element.style.width = imW + "px";
 89         }
 90         
 91         this.element.style.backgroundPosition = bposX + " " + bposY;
 92     },
 93     
 94     setMaxTiles : function (mtx, mty) {
 95     }
 96 }
 97 
 98 bigshot.object.validate ("bigshot.HTMLDivElementLayer", bigshot.Layer);
