 17 /**
 18  * Creates a new image layer.
 19  * 
 20  * @param {bigshot.ImageBase} image the image that this layer is part of
 21  * @param {bigshot.ImageParameters} parameters the associated image parameters
 22  * @param {number} w the current width in css pixels of the viewport
 23  * @param {number} h the current height in css pixels of the viewport
 24  * @param {bigshot.ImageTileCache} itc the tile cache to use
 25  * @class A tiled, zoomable image layer.
 26  * @constructor
 27  */
 28 bigshot.TileLayer = function (image, parameters, w, h, itc) {
 29     this.rows = new Array ();
 30     this.browser = new bigshot.Browser ();
 31     this.container = image.createLayerContainer ();
 32     this.parentContainer = image.getContainer ();
 33     this.parameters = parameters;
 34     this.w = w;
 35     this.h = h;
 36     this.imageTileCache = itc;
 37     
 38     this.resize (w, h);
 39     return this;
 40 }
 41 
 42 bigshot.TileLayer.prototype = {
 43     getContainer : function () {
 44         return this.container;
 45     },
 46     
 47     resize : function (w, h) {
 48         this.container.style.width = this.parentContainer.clientWidth + "px";
 49         this.container.style.height = this.parentContainer.clientHeight + "px";
 50         this.pixelWidth = this.parentContainer.clientWidth;
 51         this.pixelHeight = this.parentContainer.clientHeight;
 52         this.w = w;
 53         this.h = h;
 54         this.rows = new Array ();
 55         this.browser.removeAllChildren (this.container);
 56         for (var r = 0; r < h; ++r) {
 57             var row = new Array ();
 58             for (var c = 0; c < w; ++c) {
 59                 var tileAnchor = document.createElement ("div");
 60                 tileAnchor.style.position = "absolute";
 61                 tileAnchor.style.overflow = "hidden";
 62                 tileAnchor.style.width = this.container.clientWidth + "px";
 63                 tileAnchor.style.height = this.container.clientHeight + "px";
 64                 
 65                 var tile = document.createElement ("div");
 66                 tile.style.position = "relative";
 67                 tile.style.border = "hidden";
 68                 tile.style.visibility = "hidden";
 69                 tile.bigshotData = {
 70                     visible : false
 71                 };
 72                 row.push (tile);
 73                 this.container.appendChild (tileAnchor);
 74                 tileAnchor.appendChild (tile);
 75             }
 76             this.rows.push (row);
 77         }
 78     },
 79     
 80     layout : function (zoom, x0, y0, tx0, ty0, size, stride, opacity) {
 81         zoom = Math.min (0, Math.ceil (zoom));
 82 
 83         this.imageTileCache.resetUsed ();
 84         var y = y0;
 85         
 86         var visible = 0;
 87         for (var r = 0; r < this.h; ++r) {
 88             var x = x0;
 89             for (var c = 0; c < this.w; ++c) {
 90                 var tile = this.rows[r][c];
 91                 var bigshotData = tile.bigshotData;
 92                 if (x + size < 0 || x > this.pixelWidth || y + size < 0 || y > this.pixelHeight) {
 93                     if (bigshotData.visible) {
 94                         bigshotData.visible = false;
 95                         tile.style.visibility = "hidden";
 96                     }
 97                 } else {
 98                     visible++;
 99                     tile.style.left = x + "px";
100                     tile.style.top = y + "px";
101                     tile.style.width = size + "px";
102                     tile.style.height = size + "px";
103                     tile.style.opacity = opacity;
104                     if (!bigshotData.visible) {
105                         bigshotData.visible = true;
106                         tile.style.visibility = "visible";
107                     }
108                     var tx = c + tx0;
109                     var ty = r + ty0;
110                     if (this.parameters.wrapX) {
111                         if (tx < 0 || tx >= this.imageTileCache.maxTileX) {
112                             tx = (tx + this.imageTileCache.maxTileX) % this.imageTileCache.maxTileX;
113                         }
114                     }
115                     
116                     if (this.parameters.wrapY) {
117                         if (ty < 0 || ty >= this.imageTileCache.maxTileY) {
118                             ty = (ty + this.imageTileCache.maxTileY) % this.imageTileCache.maxTileY;
119                         }
120                     }
121                     
122                     var imageKey = tx + "_" + ty + "_" + zoom;
123                     var isOutside = tx < 0 || tx >= this.imageTileCache.maxTileX || ty < 0 || ty >= this.imageTileCache.maxTileY;
124                     if (isOutside) {
125                         if (!bigshotData.isOutside) {
126                             var image = this.imageTileCache.getImage (tx, ty, zoom);
127                             
128                             this.browser.removeAllChildren (tile);
129                             tile.appendChild (image);
130                             bigshotData.image = image;
131                         }
132                         bigshotData.isOutside = true;
133                         bigshotData.imageKey = "EMPTY";
134                         bigshotData.image.style.width = size + "px";
135                         bigshotData.image.style.height = size + "px";                            
136                     } else {
137                         var image = this.imageTileCache.getImage (tx, ty, zoom);
138                         
139                         bigshotData.isOutside = false;
140                         
141                         if (bigshotData.imageKey !== imageKey || bigshotData.isPartial) {
142                             this.browser.removeAllChildren (tile);
143                             tile.appendChild (image);
144                             bigshotData.image = image;
145                             bigshotData.imageKey = imageKey;     
146                             bigshotData.isPartial = image.isPartial;
147                         }
148                         bigshotData.image.style.width = size + "px";
149                         bigshotData.image.style.height = size + "px";
150                         
151                     }                    
152                 }
153                 x += stride;
154             }
155             y += stride;
156         }
157     },
158     
159     setMaxTiles : function (mtx, mty) {
160         this.imageTileCache.setMaxTiles (mtx, mty);
161     }
162 };
163 
164 bigshot.object.validate ("bigshot.TileLayer", bigshot.Layer);
