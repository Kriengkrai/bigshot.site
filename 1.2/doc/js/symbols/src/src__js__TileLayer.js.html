 17 /**
 18  * Creates a new image layer.
 19  * 
 20  * @param {bigshot.Image} image the image that this layer is part of
 21  * @param {bigshot.ImageParameters} parameters the associated image parameters
 22  * @param {number} w the current width in css pixels of the viewport
 23  * @param {number} h the current height in css pixels of the viewport
 24  * @param {bigshot.ImageTileCache} itc the tile cache to use
 25  * @class A tiled, zoomable image layer.
 26  * @constructor
 27  */
 28 bigshot.TileLayer = function (image, parameters, w, h, itc) {
 29     this.rows = new Array ();
 30     this.browser = new bigshot.Browser ();
 31     this.container = image.createLayerContainer ();
 32     this.parentContainer = image.getContainer ();
 33     this.parameters = parameters;
 34     this.w = w;
 35     this.h = h;
 36     this.imageTileCache = itc;
 37     
 38     this.resize (w, h);
 39     return this;
 40 }
 41  
 42 bigshot.TileLayer.prototype = {
 43     getContainer : function () {
 44         return this.container;
 45     },
 46     
 47     resize : function (w, h) {
 48         this.container.style.width = this.parentContainer.clientWidth + "px";
 49         this.container.style.height = this.parentContainer.clientHeight + "px";
 50         
 51         this.w = w;
 52         this.h = h;
 53         this.rows = new Array ();
 54         this.browser.removeAllChildren (this.container);
 55         for (var r = 0; r < h; ++r) {
 56             var row = new Array ();
 57             for (var c = 0; c < w; ++c) {
 58                 var tileAnchor = document.createElement ("div");
 59                 tileAnchor.style.position = "absolute";
 60                 tileAnchor.style.overflow = "hidden";
 61                 tileAnchor.style.width = this.container.clientWidth + "px";
 62                 tileAnchor.style.height = this.container.clientHeight + "px";
 63                 
 64                 var tile = document.createElement ("div");
 65                 tile.style.position = "relative";
 66                 tile.style.visibility = "hidden";
 67                 tile.style.border = "hidden";
 68                 row.push (tile);
 69                 this.container.appendChild (tileAnchor);
 70                 tileAnchor.appendChild (tile);
 71             }
 72             this.rows.push (row);
 73         }
 74     },
 75     
 76     layout : function (zoom, x0, y0, tx0, ty0, size, stride, opacity) {
 77         zoom = Math.min (0, Math.ceil (zoom));
 78         this.imageTileCache.resetUsed ();
 79         var y = y0;
 80         for (var r = 0; r < this.h; ++r) {
 81             var x = x0;
 82             for (var c = 0; c < this.w; ++c) {
 83                 var tile = this.rows[r][c];
 84                 tile.style.left = x + "px";
 85                 tile.style.top = y + "px";
 86                 tile.style.visibility = "";
 87                 tile.style.width = size + "px";
 88                 tile.style.height = size + "px";
 89                 tile.style.opacity = opacity;
 90                 this.browser.removeAllChildren (tile);
 91                 var tx = c + tx0;
 92                 var ty = r + ty0;
 93                 if (this.parameters.wrapX) {
 94                     if (tx < 0 || tx >= this.imageTileCache.maxTileX) {
 95                         tx = (tx + this.imageTileCache.maxTileX) % this.imageTileCache.maxTileX;
 96                     }
 97                 }
 98                 
 99                 if (this.parameters.wrapY) {
100                     if (ty < 0 || ty >= this.imageTileCache.maxTileY) {
101                         ty = (ty + this.imageTileCache.maxTileY) % this.imageTileCache.maxTileY;
102                     }
103                 }
104                 var image = this.imageTileCache.getImage (tx, ty, zoom);
105                 image.style.width = size + "px";
106                 image.style.height = size + "px";
107                 tile.appendChild (image);
108                 x += stride;
109             }
110             y += stride;
111         }
112     },
113     
114     setMaxTiles : function (mtx, mty) {
115         this.imageTileCache.setMaxTiles (mtx, mty);
116     }
117 };
118 
119 bigshot.object.validate ("bigshot.TileLayer", bigshot.Layer);
