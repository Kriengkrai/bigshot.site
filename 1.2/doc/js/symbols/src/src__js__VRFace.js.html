 17 /**
 18  * Creates a new VR cube face.
 19  *
 20  * @class a VR cube face. The {@link bigshot.VRPanorama} instance holds
 21  * six of these.
 22  *
 23  * @param {bigshot.VRPanorama} owner the VR panorama this face is part of.
 24  * @param {String} key the identifier for the face. "f" is front, "b" is back, "u" is
 25  * up, "d" is down, "l" is left and "r" is right.
 26  * @param {point} topLeft_ the top-left corner of the quad.
 27  * @param {number} width_ the length of the sides of the face, expressed in multiples of u and v.
 28  * @param {vector} u basis vector going from the top left corner along the top edge of the face
 29  * @param {vector} v basis vector going from the top left corner along the left edge of the face
 30  */
 31 bigshot.VRFace = function (owner, key, topLeft_, width_, u, v, onLoaded) {
 32     var that = this;
 33     this.owner = owner;
 34     this.key = key;
 35     this.topLeft = topLeft_;
 36     this.width = width_;
 37     this.u = u;
 38     this.v = v;
 39     this.updated = false;
 40     this.parameters = new Object ();
 41     
 42     for (var k in this.owner.getParameters ()) {
 43         this.parameters[k] = this.owner.getParameters ()[k];
 44     }
 45     
 46     bigshot.setupFileSystem (this.parameters);
 47     this.parameters.fileSystem.setPrefix ("face_" + key);
 48     this.parameters.merge (this.parameters.fileSystem.getDescriptor (), false);
 49     
 50     
 51     /**
 52      * Texture cache.
 53      *
 54      * @private
 55      */
 56     this.tileCache = owner.renderer.createTileCache (function () { 
 57             that.updated = true;
 58             owner.renderUpdated (bigshot.VRPanorama.ONRENDER_TEXTURE_UPDATE);
 59         }, onLoaded, this.parameters);
 60     
 61     this.fullSize = this.parameters.width;
 62     this.overlap = this.parameters.overlap;
 63     this.tileSize = this.parameters.tileSize;
 64     
 65     this.minDivisions = 0;
 66     var fullZoom = Math.log (this.fullSize - this.overlap) / Math.LN2;
 67     var singleTile = Math.log (this.tileSize - this.overlap) / Math.LN2;
 68     this.maxDivisions = Math.floor (fullZoom - singleTile);
 69     this.maxTesselation = this.parameters.maxTesselation >= 0 ? this.parameters.maxTesselation : this.maxDivisions;
 70 }
 71 
 72 bigshot.VRFace.prototype = {
 73     browser : new bigshot.Browser (),
 74     
 75     /**
 76      * Utility function to do a multiply-and-add of a 3d point.
 77      *
 78      * @private
 79      * @param p {point} the point to multiply
 80      * @param m {number} the number to multiply the elements of p with
 81      * @param a {point} the point to add
 82      * @return p * m + a
 83      */
 84     pt3dMultAdd : function (p, m, a) {
 85         return {
 86             x : p.x * m + a.x,
 87             y : p.y * m + a.y,
 88             z : p.z * m + a.z
 89         };
 90     },
 91     
 92     /**
 93      * Utility function to do an element-wise multiply of a 3d point.
 94      *
 95      * @private
 96      * @param p {point} the point to multiply
 97      * @param m {number} the number to multiply the elements of p with
 98      * @return p * m
 99      */
100     pt3dMult : function (p, m) {
101         return {
102             x : p.x * m,
103             y : p.y * m,
104             z : p.z * m
105         };
106     },
107     
108     /**
109      * Creates a textured quad.
110      *
111      * @private
112      */
113     generateFace : function (scene, topLeft, width, tx, ty, divisions) {
114         width *= this.tileSize / (this.tileSize - this.overlap);
115         var texture = this.tileCache.getTexture (tx, ty, -this.maxDivisions + divisions);
116         scene.addQuad (this.owner.renderer.createTexturedQuad (
117                 topLeft,
118                 this.pt3dMult (this.u, width),
119                 this.pt3dMult (this.v, width),
120                 texture
121             )
122         );
123     },
124     
125     VISIBLE_NONE : 0,
126     VISIBLE_SOME : 1,
127     VISIBLE_ALL : 2,
128     
129     /**
130      * Tests whether the point is in the axis-aligned rectangle.
131      * 
132      * @private
133      * @param point the point
134      * @param min top left corner of the rectangle
135      * @param max bottom right corner of the rectangle
136      */
137     pointInRect : function (point, min, max) {
138         return (point.x >= min.x && point.y >= min.y && point.x < max.x && point.y < max.y);
139     },
140     
141     /**
142      * Intersects a quadrilateral with the view frustum.
143      * The test is a simple rectangle intersection of the AABB of
144      * the transformed quad with the viewport.
145      *
146      * @private
147      * @return VISIBLE_NONE, VISIBLE_SOME or VISIBLE_ALL
148      */
149     intersectWithView : function intersectWithView (transformed) {
150         var numNull = 0;
151         var tf = [];
152         var tfl = transformed.length;
153         for (var i = 0; i < tfl; ++i) {
154             if (transformed[i] == null) {
155                 numNull++;
156             } else {
157                 tf.push (transformed[i]);
158             }
159         }
160         if (numNull == 4) {
161             return this.VISIBLE_NONE;
162         }
163         
164         var minX = tf[0].x;
165         var minY = tf[0].y;
166         
167         var maxX = minX;
168         var maxY = minY;
169         
170         var viewMinX = 0;
171         var viewMinY = 0;
172         
173         var viewMaxX = this.viewportWidth;
174         var viewMaxY = this.viewportHeight;
175         
176         var pointsInViewport = 0;
177         var tl = tf.length;
178         for (var i = 1; i < tl; ++i) {
179             var tix = tf[i].x;
180             var tiy = tf[i].y;
181             
182             minX = minX < tix ? minX : tix;
183             minY = minY < tiy ? minY : tiy;
184             
185             
186             maxX = maxX > tix ? maxX : tix;
187             maxY = maxY > tiy ? maxY : tiy;
188         }
189         
190         var iminX = minX > viewMinX ? minX : viewMinX;
191         var iminY = minY > viewMinY ? minY : viewMinY;
192         
193         var imaxX = maxX < viewMaxX ? maxX : viewMaxX;
194         var imaxY = maxY < viewMaxY ? maxY : viewMaxY;
195         
196         if (iminX <= imaxX && iminY <= imaxY) {
197             return this.VISIBLE_SOME;
198         }            
199         
200         return this.VISIBLE_NONE;
201     },
202     
203     /**
204      * Quick and dirty computation of the on-screen distance in pixels
205      * between two 2d points. We use the max of the x and y differences.
206      * In case a point is null (that is, it's not on the screen), we 
207      * return an arbitrarily high number.
208      *
209      * @private
210      */
211     screenDistance : function screenDistance (p0, p1) {
212         if (p0 == null || p1 == null) {
213             return 0;
214         }
215         return Math.max (Math.abs (p0.x - p1.x), Math.abs (p0.y - p1.y));
216     },
217     
218     transformToScreen : function transformToScreen (v) {
219         return this.owner.renderer.transformToScreen ([v.x, v.y, v.z, 1.0]);
220     },
221     
222     /**
223      * Optionally subdivides a quad into fourn new quads, depending on the
224      * position and on-screen size of the quad.
225      *
226      * @private
227      * @param {bigshot.WebGLTexturedQuadScene} scene the scene to add quads to
228      * @param {point} topLeft the top left corner of this quad
229      * @param {number} width the sides of the quad, expressed in multiples of u and v
230      * @param {int} divisions the current number of divisions done (increases by one for each
231      * split-in-four).
232      * @param {int} tx the tile column this face is in
233      * @param {int} ty the tile row this face is in 
234      */
235     generateSubdivisionFace : function generateSubdivisionFace (scene, topLeft, width, divisions, tx, ty, transformed) {
236         if (!transformed) {
237             transformed = new Array (4);
238             transformed[0] = this.transformToScreen (topLeft);
239             var topRight = this.pt3dMultAdd (this.u, width, topLeft);
240             transformed[1] = this.transformToScreen (topRight);
241             
242             var bottomLeft = this.pt3dMultAdd (this.v, width, topLeft);
243             transformed[3] = this.transformToScreen (bottomLeft);
244             
245             var bottomRight = this.pt3dMultAdd (this.v, width, topRight);
246             transformed[2] = this.transformToScreen (bottomRight);            
247         };
248         
249         var numVisible = this.intersectWithView (transformed);
250         
251         if (numVisible == this.VISIBLE_NONE) {
252             return;
253         }
254         
255         var dmax = 0;
256         for (var i = 0; i < transformed.length; ++i) {
257             var next = (i + 1) % 4;
258             dmax = Math.max (this.screenDistance (transformed[i], transformed[next]), dmax);
259         }
260         
261         // Convert the distance to physical pixels
262         dmax *= this.owner.browser.getDevicePixelScale ();
263         
264         if (divisions < this.minDivisions 
265                 || 
266                 (
267                     (
268                         dmax > this.owner.maxTextureMagnification * (this.tileSize - this.overlap) 
269                     ) && divisions < this.maxDivisions && divisions < this.maxTesselation
270                 )
271             ) {
272                 var center = this.pt3dMultAdd ({x: this.u.x + this.v.x, y: this.u.y + this.v.y, z: this.u.z + this.v.z }, width / 2, topLeft);
273                 var midTop = this.pt3dMultAdd (this.u, width / 2, topLeft);
274                 var midLeft = this.pt3dMultAdd (this.v, width / 2, topLeft);
275                 
276                 var tCenter = this.transformToScreen (center);
277                 var tMidLeft = this.transformToScreen (midLeft);
278                 var tMidTop = this.transformToScreen (midTop);
279                 var tMidRight = this.transformToScreen (this.pt3dMultAdd (this.u, width, midLeft));
280                 var tMidBottom = this.transformToScreen (this.pt3dMultAdd (this.v, width, midTop));
281                 
282                 this.generateSubdivisionFace (scene, topLeft, width / 2, divisions + 1, tx * 2, ty * 2, [transformed[0], tMidTop, tCenter, tMidLeft]);
283                 this.generateSubdivisionFace (scene, midTop, width / 2, divisions + 1, tx * 2 + 1, ty * 2, [tMidTop, transformed[1], tMidRight, tCenter]);
284                 this.generateSubdivisionFace (scene, midLeft, width / 2, divisions + 1, tx * 2, ty * 2 + 1, [tMidLeft, tCenter, tMidBottom, transformed[3]]);
285                 this.generateSubdivisionFace (scene, center, width / 2, divisions + 1, tx * 2 + 1, ty * 2 + 1, [tCenter, tMidRight, transformed[2], tMidBottom]);
286             } else {
287                 this.generateFace (scene, topLeft, width, tx, ty, divisions);
288             }
289     },
290     
291     /**
292      * Tests if the face has had any updated texture
293      * notifications from the tile cache.
294      *
295      * @public
296      */
297     isUpdated : function () {
298         return this.updated;
299     },
300     
301     /**
302      * Renders this face into a scene.
303      * 
304      * @public
305      * @param {bigshot.WebGLTexturedQuadScene} scene the scene to render into
306      */
307     render : function (scene) {
308         this.updated = false;
309         this.viewportWidth = this.owner.renderer.getViewportWidth ();
310         this.viewportHeight = this.owner.renderer.getViewportHeight ();        
311         this.generateSubdivisionFace (scene, this.topLeft, this.width, 0, 0, 0);
312     },
313     
314     /**
315      * Performs post-render cleanup.
316      */
317     endRender : function () {
318         this.tileCache.purge ();
319     }
320 }
