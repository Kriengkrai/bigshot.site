 17 /**
 18  * @class WebGL renderer.
 19  */
 20 bigshot.WebGLVRRenderer = function (container) {
 21     this.container = container;
 22     this.canvas = document.createElement ("canvas");
 23     this.canvas.width = 480;
 24     this.canvas.height = 480;
 25     this.canvas.style.position = "absolute";
 26     this.container.appendChild (this.canvas);
 27     this.webGl = new bigshot.WebGL (this.canvas);
 28     this.webGl.initShaders();
 29     this.webGl.gl.clearColor(0.0, 0.0, 0.0, 1.0);
 30     this.webGl.gl.blendFunc (this.webGl.gl.ONE, this.webGl.gl.ZERO);
 31     this.webGl.gl.enable (this.webGl.gl.BLEND);
 32     this.webGl.gl.disable(this.webGl.gl.DEPTH_TEST);
 33     this.webGl.gl.clearDepth(1.0);
 34     
 35     var that = this;
 36     this.buffers = new bigshot.TimedWeakReference (function () {
 37             return that.setupBuffers ();
 38         }, function (buffers) {
 39             that.disposeBuffers (buffers);
 40         }, 1000);
 41 }
 42 
 43 bigshot.WebGLVRRenderer.prototype = {
 44     createTileCache : function (onloaded, onCacheInit, parameters) {
 45         return new bigshot.TextureTileCache (onloaded, onCacheInit, parameters, this.webGl);
 46     },
 47     
 48     createTexturedQuadScene : function () {
 49         return new bigshot.WebGLTexturedQuadScene (this.webGl, this.buffers);
 50     },
 51     
 52     setupBuffers : function () {
 53         var vertexPositionBuffer = this.webGl.gl.createBuffer();
 54         
 55         var textureCoordBuffer = this.webGl.gl.createBuffer();
 56         this.webGl.gl.bindBuffer(this.webGl.gl.ARRAY_BUFFER, textureCoordBuffer);
 57         var textureCoords = [
 58             // Front face
 59             0.0,  0.0,
 60             1.0,  0.0,
 61             1.0,  1.0,
 62             0.0,  1.0
 63         ];
 64         this.webGl.gl.bufferData (this.webGl.gl.ARRAY_BUFFER, new Float32Array (textureCoords), this.webGl.gl.STATIC_DRAW);
 65         
 66         var vertexIndexBuffer = this.webGl.gl.createBuffer();
 67         this.webGl.gl.bindBuffer(this.webGl.gl.ELEMENT_ARRAY_BUFFER, vertexIndexBuffer);            
 68         var vertexIndexes = [
 69             0, 2, 1,
 70             0, 3, 2
 71         ];
 72         this.webGl.gl.bufferData(this.webGl.gl.ELEMENT_ARRAY_BUFFER, new Uint16Array (vertexIndexes), this.webGl.gl.STATIC_DRAW);
 73         
 74         this.webGl.gl.bindBuffer(this.webGl.gl.ARRAY_BUFFER, textureCoordBuffer);
 75         this.webGl.gl.vertexAttribPointer(this.webGl.shaderProgram.textureCoordAttribute, 2, this.webGl.gl.FLOAT, false, 0, 0);
 76         
 77         this.webGl.gl.bindBuffer(this.webGl.gl.ARRAY_BUFFER, vertexPositionBuffer);
 78         this.webGl.gl.vertexAttribPointer(this.webGl.shaderProgram.vertexPositionAttribute, 3, this.webGl.gl.FLOAT, false, 0, 0);
 79         
 80         return {
 81             vertexPositionBuffer : vertexPositionBuffer,
 82             textureCoordBuffer : textureCoordBuffer,
 83             vertexIndexBuffer : vertexIndexBuffer
 84         };
 85     },
 86     
 87     dispose : function () {
 88         this.buffers.dispose ();
 89     },
 90     
 91     disposeBuffers : function (buffers) {
 92         this.webGl.gl.deleteBuffer (buffers.vertexPositionBuffer);
 93         this.webGl.gl.deleteBuffer (buffers.vertexIndexBuffer);
 94         this.webGl.gl.deleteBuffer (buffers.textureCoordBuffer);
 95     },
 96     
 97     getElement : function () {
 98         return this.canvas;
 99     },
100     
101     supportsUpdate : function () {
102         return false;
103     },
104     
105     createTexturedQuad : function (p, u, v, texture) {
106         return new bigshot.WebGLTexturedQuad (p, u, v, texture);
107     },
108     
109     getViewportWidth : function () {
110         return this.webGl.gl.viewportWidth;
111     },
112     
113     getViewportHeight : function () {
114         return this.webGl.gl.viewportHeight;
115     },
116     
117     beginRender : function (rotation, fov, translation, rotationOffsets) {
118         this.webGl.gl.viewport (0, 0, this.webGl.gl.viewportWidth, this.webGl.gl.viewportHeight);
119         
120         this.webGl.pMatrix.reset ();
121         this.webGl.pMatrix.perspective (fov, this.webGl.gl.viewportWidth / this.webGl.gl.viewportHeight, 0.1, 100.0);
122         
123         this.webGl.mvMatrix.reset ();
124         this.webGl.mvMatrix.translate (translation);
125         this.webGl.mvMatrix.rotateZ (rotationOffsets.r);
126         this.webGl.mvMatrix.rotateX (rotationOffsets.p);
127         this.webGl.mvMatrix.rotateY (rotationOffsets.y);
128         this.webGl.mvMatrix.rotateY (rotation.y);
129         this.webGl.mvMatrix.rotateX (rotation.p);
130         
131         this.mvMatrix = this.webGl.mvMatrix;
132         this.pMatrix = this.webGl.pMatrix;
133         this.mvpMatrix = this.pMatrix.matrix ().multiply (this.mvMatrix.matrix ());
134     },
135     
136     endRender : function () {
137         
138     },
139     
140     resize : function (w, h) {
141         this.canvas.width = w;
142         this.canvas.height = h;
143         if (this.container.style.width != "") {
144             this.container.style.width = w + "px";
145         }
146         if (this.container.style.height != "") {
147             this.container.style.height = h + "px";
148         }
149     },
150     
151     onresize : function () {
152         this.webGl.onresize ();
153     }
154 }

156 bigshot.Object.extend (bigshot.WebGLVRRenderer, bigshot.AbstractVRRenderer);
