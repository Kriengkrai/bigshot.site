 17 /**
 18  * @class WebGL renderer.
 19  */
 20 bigshot.WebGLVRRenderer = function (container) {
 21     this.container = container;
 22     this.canvas = document.createElement ("canvas");
 23     this.canvas.width = 480;
 24     this.canvas.height = 480;
 25     this.canvas.style.position = "absolute";
 26     this.container.appendChild (this.canvas);
 27     this.webGl = new bigshot.WebGL (this.canvas);
 28     this.webGl.initShaders();
 29     this.webGl.gl.clearColor(0.0, 0.0, 0.0, 1.0);
 30     this.webGl.gl.blendFunc (this.webGl.gl.ONE, this.webGl.gl.ZERO);
 31     this.webGl.gl.enable (this.webGl.gl.BLEND);
 32     this.webGl.gl.disable(this.webGl.gl.DEPTH_TEST);
 33     this.webGl.gl.clearDepth(1.0);
 34 }
 35     
 36 bigshot.WebGLVRRenderer.prototype = {
 37     createTileCache : function (onloaded, onCacheInit, parameters) {
 38         return new bigshot.TextureTileCache (onloaded, onCacheInit, parameters, this.webGl);
 39     },
 40     
 41     createTexturedQuadScene : function () {
 42         return new bigshot.WebGLTexturedQuadScene (this.webGl);
 43     },
 44     
 45     getElement : function () {
 46         return this.canvas;
 47     },
 48     
 49     supportsUpdate : function () {
 50         return false;
 51     },
 52     
 53     createTexturedQuad : function (p, u, v, texture) {
 54         return new bigshot.WebGLTexturedQuad (p, u, v, texture);
 55     },
 56     
 57     getViewportWidth : function () {
 58         return this.webGl.gl.viewportWidth;
 59     },
 60     
 61     getViewportHeight : function () {
 62         return this.webGl.gl.viewportHeight;
 63     },
 64     
 65     beginRender : function (y, p, fov, tx, ty, tz, oy, op, or) {
 66         this.webGl.gl.viewport (0, 0, this.webGl.gl.viewportWidth, this.webGl.gl.viewportHeight);
 67         
 68         this.webGl.pMatrix.reset ();
 69         this.webGl.pMatrix.perspective (fov, this.webGl.gl.viewportWidth / this.webGl.gl.viewportHeight, 0.1, 100.0);
 70         
 71         this.webGl.mvMatrix.reset ();
 72         this.webGl.mvMatrix.translate ([tx, ty, tz]);
 73         this.webGl.mvMatrix.rotate (or, [0, 0, 1]);
 74         this.webGl.mvMatrix.rotate (op, [1, 0, 0]);
 75         this.webGl.mvMatrix.rotate (oy, [0, 1, 0]);
 76         this.webGl.mvMatrix.rotate (y, [0, 1, 0]);
 77         this.webGl.mvMatrix.rotate (p, [1, 0, 0]);
 78         
 79         this.mvMatrix = this.webGl.mvMatrix;
 80         this.pMatrix = this.webGl.pMatrix;
 81         this.mvpMatrix = this.pMatrix.matrix ().multiply (this.mvMatrix.matrix ());
 82     },
 83     
 84     endRender : function () {
 85         
 86     },
 87     
 88     resize : function (w, h) {
 89         this.canvas.width = w;
 90         this.canvas.height = h;
 91         if (this.container.style.width != "") {
 92             this.container.style.width = w + "px";
 93         }
 94         if (this.container.style.height != "") {
 95             this.container.style.height = h + "px";
 96         }
 97     },
 98     
 99     onresize : function () {
100         this.webGl.onresize ();
101     }
102 }
103 
104 bigshot.object.extend (bigshot.WebGLVRRenderer, bigshot.AbstractVRRenderer);
105 bigshot.object.validate ("bigshot.WebGLVRRenderer", bigshot.VRRenderer);
