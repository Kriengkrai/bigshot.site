 59         this.mouseEnter = function (_fn) {
 60             var isAChildOf = this.isAChildOf;
 61             return function(_evt)
 62             {
 63                 var relTarget = _evt.relatedTarget;
 64                 if (this === relTarget || isAChildOf (this, relTarget))
 65                 { return; }
 66                 
 67                 _fn.call (this, _evt);
 68             }
 69         };

 71         this.isAChildOf = function (_parent, _child) {
 72             if (_parent === _child) { return false; }
 73             while (_child && _child !== _parent)
 74             { _child = _child.parentNode; }
 75             
 76             return _child === _parent;
 77         };

 83                 elem.detachEvent('on' + eventName, fn);
 84             }

 88             if (typeof _elem.addEventListener != 'undefined')
 89             {
 90                 if (_evtName === 'mouseenter')
 91                 { _elem.addEventListener('mouseover', this.mouseEnter(_fn), _useCapture); }

 93                 { _elem.addEventListener('mouseout', this.mouseEnter(_fn), _useCapture); }
 94                 else
 95                 { _elem.addEventListener(_evtName, _fn, _useCapture); }
 96             }
 97             else if (typeof _elem.attachEvent != 'undefined')
 98             {
 99                 _elem.attachEvent('on' + _evtName, _fn);
100             }
101             else
102             {

193      * @class Object-oriented support functions, used to make JavaScript
194      * a bit more palatable to a Java-head.
195      */
196     bigshot.object = {
197         /**
198          * Performs an inheritance operation with a base-class instance 
199          * and a derived instance. Each method that exists in the base class
200          * but not in the derived instance is copied across. Every method in the
201          * base-class instance is thunked and put in a field named <code>_super</code>
202          * in the derived instance. The thunk takes care of making sure the
203          * <code>this</code> reference points where you'd expect it.
204          * Methods in the derived class can refer to <code>_super.<i>methodName</i></code>
205          * to get the method as defined by the base class.
206          * Fields not of <code>function</code> type are copied across if they do not
207          * exist in the derived class.
208          *
209          * @param {Object} base the base-class instance
210          * @param {Object} derived the derived-class instance
211          */
212         extend : function (base, derived) {
213             var _super = {};
214             
215             for (var k in derived) {
216                 if (typeof (derived[k]) == "function") {
217                     derived[k] = this.makeThunk (derived[k], derived, _super);
218                 }
219             }
220             
221             for (var k in base) {
222                 if (typeof (base[k]) == "function") {
223                     var fn = base[k];
224                     var usesSuper = (fn.usesSuper ? fn.usesSuper : null);
225                     var fn = (fn.isThunkFor ? fn.isThunkFor : fn);
226                     _super[k] = this.makeThunk (fn, derived, usesSuper);
227                     if (!derived[k]) {
228                         derived[k] = _super[k];
229                     }
230                 } else if (!derived[k]) {
231                     derived[k] = base[k];
232                 }
233             }
234             return derived;
235         },
236         
237         /**
238          * Creates a function thunk that resets the <code>this</code>
239          * reference and the object's <code>_super</code> reference.
240          * The returned function has three properties that can be used
241          * to examine it:
242          *
243          * <ul>
244          * <li><code>thunksTo</code>: Set to <code>_this</code>
245          * <li><code>isThunkFor</code>: Set to <code>fn</code>
246          * <li><code>usesSuper</code>: Set to <code>_super</code>
247          * </ul>
248          * 
249          * @param {function} fn the function to thunk
250          * @param {Object} _this the new <code>this</code> reference.
251          * @param {Object} _super the new <code>_super</code> reference.
252          */
253         makeThunk : function (fn, _this, _super) {
254             var f = function () {
255                 _this._super = _super;
256                 return fn.apply (_this, arguments);
257             };
258             f.thunksTo = _this;
259             f.isThunkFor = fn;
260             f.usesSuper = _super;
261             return f;
262         },

321             this.element.style.height = sh + "px";

412      * Creates a new hotspot layer. The layer must be added to the image using
413      * {@link bigshot.Image#addLayer}.
414      *
415      * @class A hotspot layer.
416      * @example
417      * var image = new bigshot.Image (...);
418      * var hotspotLayer = new bigshot.HotspotLayer (image);
419      * var hotspot = new bigshot.LinkHotspot (100, 100, 200, 100, 
420      *    "Bigshot on Google Code", 
421      *    "http://code.google.com/p/bigshot/");
422      *
423      * // Style the hotspot a bit
424      * hotspot.getElement ().className = "hotspot"; 
425      * hotspot.getLabel ().className = "label";
426      *
427      * hotspotLayer.addHotspot (hotspot);
428      *
429      * image.addLayer (hotspotLayer);
430      * 
431      * @param {bigshot.Image} image the image this hotspot layer will be part of
432      * @augments bigshot.Layer
433      * @constructor
434      */
435     bigshot.HotspotLayer = function (image) {
436         this.hotspots = new Array ();
437         this.browser = new bigshot.Browser ();
438         this.container = image.createLayerContainer ();
439         this.parentContainer = image.getContainer ();
440         
441         this.getContainer = function () {
442             return this.container;
443         };
444         
445         this.resize = function (w, h) {
446             this.container.style.width = this.parentContainer.clientWidth + "px";
447             this.container.style.height = this.parentContainer.clientHeight + "px";
448         };
449         
450         this.layout = function (zoom, x0, y0, tx0, ty0, size, stride, opacity) {
451             var zoomFactor = Math.pow (2, zoom);
452             x0 -= stride * tx0;
453             y0 -= stride * ty0;
454             for (var i = 0; i < this.hotspots.length; ++i) {
455                 this.hotspots[i].layout (x0, y0, zoomFactor);
456             }            
457         };
458         
459         this.setMaxTiles = function (mtx, mty) {
460         };
461         
462         /**
463          * Adds a hotspot to the layer. 
464          *
465          * @param {bigshot.Hotspot} hotspot the hotspot to add.
466          */
467         this.addHotspot = function (hotspot) {
468             this.container.appendChild (hotspot.getElement ());
469             this.hotspots.push (hotspot);
470         };
471         
472         this.resize (0, 0);
473         return bigshot.object.extend (new bigshot.Layer (), this);
474     }

476     /**
477      * Creates a new image layer.
478      * 
479      * @param {bigshot.Image} image the image that this layer is part of
480      * @param {bigshot.ImageParameters} parameters the associated image parameters
481      * @param {number} w the current width in css pixels of the viewport
482      * @param {number} h the current height in css pixels of the viewport
483      * @param {bigshot.ImageTileCache} itc the tle cache to use
484      * @class A tiled, zoomable image layer.
485      * @constructor
486      */
487     bigshot.TileLayer = function (image, parameters, w, h, itc) {
488         this.rows = new Array ();
489         this.browser = new bigshot.Browser ();
490         this.container = image.createLayerContainer ();
491         this.parentContainer = image.getContainer ();
492         this.parameters = parameters;
493         this.w = w;
494         this.h = h;
495         this.imageTileCache = itc;
496         
497         this.getContainer = function () {
498             return this.container;
499         };
500         
501         this.resize = function (w, h) {
502             this.container.style.width = this.parentContainer.clientWidth + "px";
503             this.container.style.height = this.parentContainer.clientHeight + "px";
504             
505             this.w = w;
506             this.h = h;
507             this.rows = new Array ();
508             this.browser.removeAllChildren (this.container);
509             for (var r = 0; r < h; ++r) {
510                 var row = new Array ();
511                 for (var c = 0; c < w; ++c) {
512                     var tileAnchor = document.createElement ("div");
513                     tileAnchor.style.position = "absolute";
514                     tileAnchor.style.overflow = "hidden";
515                     tileAnchor.style.width = this.container.clientWidth + "px";
516                     tileAnchor.style.height = this.container.clientHeight + "px";
517                     
518                     var tile = document.createElement ("div");
519                     tile.style.position = "relative";
520                     tile.style.visibility = "hidden";
521                     tile.style.border = "hidden";
522                     row.push (tile);
523                     this.container.appendChild (tileAnchor);
524                     tileAnchor.appendChild (tile);
525                 }
526                 this.rows.push (row);
527             }
528         };
529         
530         this.layout = function (zoom, x0, y0, tx0, ty0, size, stride, opacity) {
531             zoom = Math.min (0, Math.ceil (zoom));
532             this.imageTileCache.resetUsed ();
533             var y = y0;
534             for (var r = 0; r < this.h; ++r) {
535                 var x = x0;
536                 for (var c = 0; c < this.w; ++c) {
537                     var tile = this.rows[r][c];
538                     tile.style.left = x + "px";
539                     tile.style.top = y + "px";
540                     tile.style.visibility = "";
541                     tile.style.width = size + "px";
542                     tile.style.height = size + "px";
543                     tile.style.opacity = opacity;
544                     this.browser.removeAllChildren (tile);
545                     var tx = c + tx0;
546                     var ty = r + ty0;
547                     if (this.parameters.wrapX) {
548                         if (tx < 0 || tx >= this.imageTileCache.maxTileX) {
549                             tx = (tx + this.imageTileCache.maxTileX) % this.imageTileCache.maxTileX;
550                         }
551                     }
552                     
553                     if (this.parameters.wrapY) {
554                         if (ty < 0 || ty >= this.imageTileCache.maxTileY) {
555                             ty = (ty + this.imageTileCache.maxTileY) % this.imageTileCache.maxTileY;
556                         }
557                     }
558                     var image = this.imageTileCache.getImage (tx, ty, zoom);
559                     image.style.width = size + "px";
560                     image.style.height = size + "px";
561                     tile.appendChild (image);
562                     x += stride;
563                 }
564                 y += stride;
565             }
566         };
567         
568         this.setMaxTiles = function (mtx, mty) {
569             this.imageTileCache.setMaxTiles (mtx, mty);
570         };
571         this.resize (w, h);
572         return this;
573     };

575     /**
576      * Creates a new, empty, LRUMap instance.
577      * 
578      * @class Implementation of a Least-Recently-Used cache map.
579      * Used by the ImageTileCache to keep track of cache entries.
580      * @constructor
581      */
582     bigshot.LRUMap = function () {
583         /** 
584          * Key to last-accessed time mapping.
585          *
586          * @type Object
587          */
588         this.keyToTime = {};
589         
590         /**
591          * Current time counter. Incremented for each access of
592          * a key in the map.
593          * @type int
594          */
595         this.counter = 0;
596         
597         /** 
598          * Current size of the map.
599          * @type int
600          */
601         this.size = 0;
602         
603         /**
604          * Marks access to an item, represented by its key in the map. 
605          * The key's last-accessed time is updated to the current time
606          * and the current time is incremented by one step.
607          *
608          * @param {String} key the key associated with the accessed item
609          */
610         this.access = function (key) {
611             this.remove (key);
612             this.keyToTime[key] = this.counter;
613             ++this.counter;
614             ++this.size;
615         };
616         
617         /**
618          * Removes a key from the map.
619          *
620          * @param {String} key the key to remove
621          * @returns true iff the key existed in the map.
622          * @type boolean
623          */
624         this.remove = function (key) {
625             if (this.keyToTime[key]) {
626                 delete this.keyToTime[key];
627                 --this.size;
628                 return true;
629             } else {
630                 return false;
631             }
632         };
633         
634         /**
635          * Returns the current number of keys in the map.
636          * @type int
637          */
638         this.getSize = function () {
639             return this.size;
640         };
641         
642         /**
643          * Returns the key in the map with the lowest
644          * last-accessed time. This is done as a linear
645          * search through the map. It could be done much 
646          * faster with a sorted map, but unless this becomes
647          * a bottleneck it is just not worth the effort.
648          * @type String
649          */
650         this.leastUsed = function () {
651             var least = this.counter + 1;
652             var leastKey = null;
653             for (var k in this.keyToTime) {
654                 if (this.keyToTime[k] < least) {
655                     least = this.keyToTime[k];
656                     leastKey = k;
657                 }
658             }
659             return leastKey;
660         };
661     };

712                     return null;
713                 }
714                 canvas.width = partialImageSize;
715                 canvas.height = partialImageSize;
716                 var ctx = canvas.getContext('2d'); 
717                 
718                 var posterScale = parameters.posterSize / Math.max (parameters.width, parameters.height);
719                 
720                 var posterWidth = Math.floor (posterScale * parameters.width);
721                 var posterHeight = Math.floor (posterScale * parameters.height);
722                 
723                 var tileSizeAtZoom = posterScale * parameters.tileSize / Math.pow (2, zoomLevel);    
724                 var sx = Math.floor (tileSizeAtZoom * tileX);
725                 var sy = Math.floor (tileSizeAtZoom * tileY);
726                 var sw = Math.floor (tileSizeAtZoom);
727                 var sh = Math.floor (tileSizeAtZoom);
728                 var dw = partialImageSize + 2;
729                 var dh = partialImageSize + 2;
730                 
731                 if (sx + sw > posterWidth) {
732                     sw = posterWidth - sx;
733                     dw = partialImageSize * (sw / Math.floor (tileSizeAtZoom));
734                 }
735                 if (sy + sh > posterHeight) {
736                     sh = posterHeight - sy;
737                     dh = partialImageSize * (sh / Math.floor (tileSizeAtZoom));
738                 }
739                 
740                 ctx.drawImage (this.fullImage, sx, sy, sw, sh, -1, -1, dw, dh);
741                 var tile = document.createElement ("img");
742                 tile.src = canvas.toDataURL ();
743                 return tile;
744             } else {
745                 return null;
746             }
747         };

750             var tile = document.createElement ("img");
751             if (parameters.emptyImage) {
752                 tile.src = parameters.emptyImage;
753             } else {
754                 tile.src = "data:image/gif,GIF89a%01%00%01%00%80%00%00%00%00%00%FF%FF%FF!%F9%04%00%00%00%00%00%2C%00%00%00%00%01%00%01%00%00%02%02D%01%00%3B";
755             }
756             return tile;
757         };

759         this.getImage = function (tileX, tileY, zoomLevel) {
760             if (tileX < 0 || tileY < 0 || tileX >= this.maxTileX || tileY >= this.maxTileY) {
761                 return this.getEmptyImage ();
762             }
763             
764             var key = this.getImageKey (tileX, tileY, zoomLevel);
765             this.lruMap.access (key);
766             
767             if (this.cachedImages[key]) {
768                 if (this.usedImages[key]) {
769                     var tile = document.createElement ("img");
770                     tile.src = this.getImageFilename (tileX, tileY, zoomLevel);
771                     return tile;
772                 } else {
773                     this.usedImages[key] = true;
774                     return this.cachedImages[key];
775                 }
776             } else {
777                 var img = this.getPartialImage (tileX, tileY, zoomLevel);
778                 if (img != null) {
779                     this.cachedImages[key] = img;
780                 } else {
781                     img = this.getEmptyImage ();
782                 }
783                 this.requestImage (tileX, tileY, zoomLevel);
784                 return img;
785             }
786         };
787         
788         this.requestImage = function (tileX, tileY, zoomLevel) {
789             var key = this.getImageKey (tileX, tileY, zoomLevel);
790             if (!this.requestedImages[key]) {
791                 this.imageRequests++;
792                 var tile = document.createElement ("img");
793                 var that = this;
794                 this.browser.registerListener (tile, "load", function () {                        
795                         that.cachedImages[key] = tile;
796                         delete that.requestedImages[key];
797                         that.imageRequests--;
798                         var now = new Date();
799                         if (that.imageRequests == 0 || now.getTime () > (that.lastOnLoadFiredAt + 50)) {
800                             that.purgeCache ();
801                             that.lastOnLoadFiredAt = now.getTime ();
802                             that.onLoaded ();
803                         }
804                     }, false);
805                 this.requestedImages[key] = tile;
806                 tile.src = this.getImageFilename (tileX, tileY, zoomLevel);                    
807             }            
808         };
809         
810         this.purgeCache = function () {
811             for (var i = 0; i < 4; ++i) {
812                 if (this.lruMap.getSize () > this.maxCacheSize) {
813                     var leastUsed = this.lruMap.leastUsed ();
814                     this.lruMap.remove (leastUsed);
815                     delete this.cachedImages[leastUsed];                    
816                 }
817             }
818         };

820         this.getImageKey = function (tileX, tileY, zoomLevel) {
821             return "I" + tileX + "_" + tileY + "_" + zoomLevel;
822         };

824         this.getImageFilename = function (tileX, tileY, zoomLevel) {
825             var f = parameters.fileSystem.getImageFilename (tileX, tileY, zoomLevel);
826             return f;
827         };
828         
829         return this;

1591          *
1592          * @public
1593          * @param {number} [x] the new x-coordinate
1594          * @param {number} [y] the new y-coordinate
1595          * @param {number} [zoom] the new zoom level
1596          */
1597         this.flyTo = function (x, y, zoom) {
1598             var that = this;
1599             
1600             x = x != null ? x : this.x;
1601             y = y != null ? y : this.y;
1602             zoom = zoom != null ? zoom : this.zoom;
1603             
1604             var targetX = Math.max (0, Math.min (this.width, x));
1605             var targetY = Math.max (0, Math.min (this.height, y));
1606             
1607             var targetZoom = Math.min (this.maxZoom, Math.max (zoom, this.minZoom));
1608             
1609             this.flying++;
1610             var flyingAtStart = this.flying;
1611             
1612             var approach = function (current, target, step) {
1613                 return current + (target - current) * step;
1614             };
1615             
1616             var iter = function () {
1617                 if (that.flying == flyingAtStart) {
1618                     var nx = approach (that.x, targetX, 0.5);
1619                     var ny = approach (that.y, targetY, 0.5);
1620                     var nz = approach (that.zoom, targetZoom, 0.5);
1621                     var done = true;
1622                     if (Math.abs (that.x - targetX) < 1.0) {
1623                         nx = targetX;
1624                     } else {
1625                         done = false;
1626                     }
1627                     if (Math.abs (that.y - targetY) < 1.0) {
1628                         ny = targetY;
1629                     } else {
1630                         done = false;
1631                     }
1632                     if (Math.abs (that.zoom - targetZoom) < 0.02) {
1633                         nz = targetZoom;
1634                     } else {
1635                         done = false;
1636                     }
1637                     that.setPosition (nx, ny, false);
1638                     that.setZoom (nz, false);
1639                     that.layout ();
1640                     if (!done) {
1641                         setTimeout (iter, 20);
1642                     }
1643                 };
1644             }
1645             setTimeout (iter, 20);
1646         };
